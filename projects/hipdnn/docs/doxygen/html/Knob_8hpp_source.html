208// Factory method implementation
209inline std::pair<Error, Knob> Knob::tryFromFlatbuffer(hipdnnBackendFlatbufferData_t fbData)
210{
211 if(fbData.ptr == nullptr || fbData.size == 0)
212 {
213 return {{ErrorCode::INVALID_VALUE, "Flatbuffer data is nullptr or has zero size"}, {}};
214 }
215
216 hipdnn_data_sdk::flatbuffer_utilities::KnobWrapper knobWrapper(fbData.ptr, fbData.size);
217
218 if(!knobWrapper.isValid())
219 {
220 return {{ErrorCode::INVALID_VALUE, "Knob flatbuffer failed verification"}, {}};
221 }
222
223 auto fbKnob = &knobWrapper.getKnob();
224
225 // Unpack to native KnobT - all conversions done automatically by FlatBuffers
226 std::unique_ptr<hipdnn_data_sdk::data_objects::KnobT> knobT(fbKnob->UnPack());
227
228 // Get knob_id for error messages
229 std::string knobId = knobT->knob_id;
230
231 // Extract default value from the union
232 KnobValueVariant defaultValue;
233 switch(knobT->default_value.type)
234 {
235 case hipdnn_data_sdk::data_objects::KnobValue::IntValue:
236 defaultValue = knobT->default_value.AsIntValue()->value;
237 break;
238 case hipdnn_data_sdk::data_objects::KnobValue::FloatValue:
239 defaultValue = knobT->default_value.AsFloatValue()->value;
240 break;
241 case hipdnn_data_sdk::data_objects::KnobValue::StringValue:
242 defaultValue = knobT->default_value.AsStringValue()->value; // Already std::string
243 break;
244 case hipdnn_data_sdk::data_objects::KnobValue::NONE:
245 // NONE default_value is invalid - knobs must have a default value
246 return {{ErrorCode::INVALID_VALUE,
247 "Knob '" + knobId + "' has NONE default_value - knobs must have a default value"},
248 {}};
249 default:
250 return {{ErrorCode::INVALID_VALUE, "Knob '" + knobId + "' has unknown default_value type"},
251 {}};
252 }
253
254 // Create the knob - strings are already std::string in KnobT
255 Knob knob(std::move(knobT->knob_id),
256 std::move(knobT->description),
257 std::move(defaultValue),
258 knobT->deprecated);
259
260 // Handle constraints using the native union types
261 switch(knobT->constraint.type)
262 {
263 case hipdnn_data_sdk::data_objects::KnobConstraint::IntConstraint:
264 {
265 auto* c = knobT->constraint.AsIntConstraint();
266 std::unordered_set<int64_t> validValues(c->valid_values.begin(), c->valid_values.end());
267 knob._constraint = std::make_unique<IntConstraint>(
268 c->min_value, c->max_value, c->step, std::move(validValues));
269 break;
270 }
271 case hipdnn_data_sdk::data_objects::KnobConstraint::FloatConstraint:
272 {
273 auto* c = knobT->constraint.AsFloatConstraint();
274 knob._constraint = std::make_unique<FloatConstraint>(c->min_value, c->max_value);
275 break;
276 }
277 case hipdnn_data_sdk::data_objects::KnobConstraint::StringConstraint:
278 {
279 auto* c = knobT->constraint.AsStringConstraint();
280 std::unordered_set<std::string> validValues(c->valid_values.begin(), c->valid_values.end());
281 knob._constraint
282 = std::make_unique<StringConstraint>(c->max_length, std::move(validValues));
283 break;
284 }
285 case hipdnn_data_sdk::data_objects::KnobConstraint::NONE:
286 knob._constraint = std::make_unique<EmptyConstraint>();
287 break;
288 default:
289 return {{ErrorCode::INVALID_VALUE, "Knob '" + knobId + "' has unknown constraint type"},
290 {}};
291 }
292
293 // Validate that the default_value satisfies the constraint
294 if(knob._constraint)
295 {
296 KnobSetting defaultSetting(knob._knobId, knob._defaultValue);
297 auto validationError = knob._constraint->validateKnobSetting(defaultSetting);
298 if(validationError.code != ErrorCode::OK)
299 {
300 return {{ErrorCode::INVALID_VALUE,
301 "Knob '" + knob._knobId + "' has default_value that violates its constraint: "
302 + validationError.err_msg},
303 {}};
304 }
305 }
306
307 return {{ErrorCode::OK, ""}, knob};
308}

312inline Error getKnobsForEngine(std::vector<Knob>& knobs, hipdnnBackendDescriptor_t engineDesc)
313{
314 int64_t knobCount = 0;
315 HIPDNN_RETURN_ON_BACKEND_FAILURE(
316 hipdnnBackend()->backendGetAttribute(engineDesc,
317 HIPDNN_ATTR_KNOB_INFO_SERIALIZED_VALUE_EXT,
318 HIPDNN_TYPE_FLATBUFFER_DATA_STRUCT_EXT,
319 0,
320 &knobCount,
321 nullptr),
322 "Failed to get knob count from engine descriptor.");
323
324 if(knobCount == 0)
325 {
326 knobs.clear();
327 return {ErrorCode::OK, ""};
328 }
329
330 std::vector<hipdnnBackendFlatbufferData_t> flatbufferDataArray(static_cast<size_t>(knobCount));
331
332 int64_t actualCount = 0;
333 HIPDNN_RETURN_ON_BACKEND_FAILURE(
334 hipdnnBackend()->backendGetAttribute(engineDesc,
335 HIPDNN_ATTR_KNOB_INFO_SERIALIZED_VALUE_EXT,
336 HIPDNN_TYPE_FLATBUFFER_DATA_STRUCT_EXT,
337 knobCount,
338 &actualCount,
339 flatbufferDataArray.data()),
340 "Failed to get knob flatbuffer data from engine descriptor.");
341
342 if(actualCount != knobCount)
343 {
344 return {ErrorCode::HIPDNN_BACKEND_ERROR,
345 "Mismatch between expected and actual knob count."};
346 }
347
348 knobs.clear();
349 knobs.reserve(static_cast<size_t>(actualCount));
350
351 std::unordered_set<std::string> usedKnobIds;
352 size_t skippedCount = 0;
353
354 for(size_t i = 0; i < static_cast<size_t>(actualCount); ++i)
355 {
356 auto [error, knob] = Knob::tryFromFlatbuffer(flatbufferDataArray[i]);
357
358 if(error.code != ErrorCode::OK)
359 {
360 // Log error and skip this knob - don't fail the entire operation
361 HIPDNN_FE_LOG_ERROR("Skipping invalid knob at index " << i << ": " << error.err_msg);
362 ++skippedCount;
363 continue;
364 }
365
366 // Check for duplicate knob IDs
367 if(!usedKnobIds.insert(knob.knobId()).second)
368 {
369 HIPDNN_FE_LOG_ERROR("Skipping knob with duplicate ID: " << knob.knobId());
370 ++skippedCount;
371 continue;
372 }
373
374 knobs.emplace_back(std::move(knob));
375 }
376
377 // If any knobs were skipped, include that information in the return message
378 if(skippedCount > 0)
379 {
380 std::ostringstream oss;
381 oss << "Loaded " << knobs.size() << " knobs, skipped " << skippedCount
382 << " invalid/duplicate knobs";
383 HIPDNN_FE_LOG_WARN(oss.str());
384 return {ErrorCode::OK, oss.str()};
385 }
386
387 return {ErrorCode::OK, ""};
388}
