232 flatbuffers::Offset<hipdnn_data_sdk::data_objects::BatchnormBackwardAttributes>
233 pack_attributes(flatbuffers::FlatBufferBuilder& builder) const // NOLINT
234 {
235 auto peerStatsVector = std::vector<int64_t>{};
236 for(const auto& peerStat : peer_stats)
237 {
238 if(peerStat)
239 {
240 peerStatsVector.emplace_back(peerStat->get_uid());
241 }
242 }
243
244 auto mean = get_mean();
245 auto invVariance = get_inv_variance();
246
247 return hipdnn_data_sdk::data_objects::CreateBatchnormBackwardAttributesDirect(
248 builder,
249 get_dy()->get_uid(),
250 get_x()->get_uid(),
251 mean ? flatbuffers::Optional<int64_t>(mean->get_uid()) : flatbuffers::nullopt,
252 invVariance ? flatbuffers::Optional<int64_t>(invVariance->get_uid())
253 : flatbuffers::nullopt,
254 get_scale()->get_uid(),
255 &peerStatsVector,
256 get_dx()->get_uid(),
257 get_dscale()->get_uid(),
258 get_dbias()->get_uid());
259 }

261 static BatchnormBackwardAttributes fromFlatBuffer(
262 const hipdnn_data_sdk::data_objects::BatchnormBackwardAttributes* fb,
263 const std::unordered_map<int64_t, std::shared_ptr<TensorAttributes>>& tensorMap)
264 {
265 BatchnormBackwardAttributes attr;
266
267 attr.set_dy(tensorMap.at(fb->dy_tensor_uid()));
268 attr.set_x(tensorMap.at(fb->x_tensor_uid()));
269 attr.set_scale(tensorMap.at(fb->scale_tensor_uid()));
270
271 if(fb->mean_tensor_uid().has_value())
272 {
273 attr.set_mean(tensorMap.at(fb->mean_tensor_uid().value()));
274 }
275 if(fb->inv_variance_tensor_uid().has_value())
276 {
277 attr.set_inv_variance(tensorMap.at(fb->inv_variance_tensor_uid().value()));
278 }
279
280 std::vector<std::shared_ptr<TensorAttributes>> peerStats;
281 if(fb->peer_stats_tensor_uid() != nullptr)
282 {
283 for(auto uid : *fb->peer_stats_tensor_uid())
284 {
285 peerStats.push_back(tensorMap.at(uid));
286 }
287 }
288 attr.set_peer_stats(peerStats);
289
290 attr.set_dx(tensorMap.at(fb->dx_tensor_uid()));
291 attr.set_dscale(tensorMap.at(fb->dscale_tensor_uid()));
292 attr.set_dbias(tensorMap.at(fb->dbias_tensor_uid()));
293
294 return attr;
295 }
