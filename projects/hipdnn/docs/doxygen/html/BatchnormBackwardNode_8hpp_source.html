28 Error pre_validate_node() const override
29 {
30 // ====================================================================
31 // BATCH NORMALIZATION BACKWARD VALIDATION
32 // (Spatial Mode: per-channel statistics over N×H×W for 4D, N×D×H×W for 5D)
33 // ====================================================================
34 // Algorithm Overview:
35 // Given dy (gradient of loss w.r.t. y), compute gradients w.r.t. inputs:
36 //
37 // INPUTS: x, dy, scale, mean_c, invStd_c (saved from forward pass)
38 // OUTPUTS: dx, dscale, dbias
39 //
40 // For each channel c, using saved forward stats (mean_c, invStd_c):
41 // xhat[n,c,h,w] = (x[n,c,h,w] - mean_c) * invStd_c
42 //
43 // Compute parameter gradients (accumulated over N,H,W):
44 // dscale_c += xhat[n,c,h,w] * dy[n,c,h,w] // gradient of scale
45 // dbias_c += dy[n,c,h,w] // gradient of bias
46 //
47 // Compute input gradient (where m = N*H*W per channel):
48 // dx_i = (scale_c * invStd_c / m) * (m*dy_i - dbias_c - xhat_i*dscale_c)
49 //
50 // This chain rule derivative accounts for batch statistics coupling.
51 // ====================================================================
52
53 // SECTION 1: Validate Required Tensor Pointers
54 HIPDNN_RETURN_IF_FALSE(attributes.get_dy(),
55 ErrorCode::ATTRIBUTE_NOT_SET,
56 "BatchnormBackwardNode missing dy for pre-validation");
57
58 HIPDNN_RETURN_IF_FALSE(attributes.get_x(),
59 ErrorCode::ATTRIBUTE_NOT_SET,
60 "BatchnormBackwardNode missing x for pre-validation");
61
62 HIPDNN_RETURN_IF_FALSE(attributes.get_scale(),
63 ErrorCode::ATTRIBUTE_NOT_SET,
64 "BatchnormBackwardNode missing scale for pre-validation");
65
66 HIPDNN_RETURN_IF_FALSE(attributes.get_dx(),
67 ErrorCode::ATTRIBUTE_NOT_SET,
68 "BatchnormBackwardNode missing dx for pre-validation");
69
70 HIPDNN_RETURN_IF_FALSE(attributes.get_dscale(),
71 ErrorCode::ATTRIBUTE_NOT_SET,
72 "BatchnormBackwardNode missing dscale for pre-validation");
73
74 HIPDNN_RETURN_IF_FALSE(attributes.get_dbias(),
75 ErrorCode::ATTRIBUTE_NOT_SET,
76 "BatchnormBackwardNode missing dbias for pre-validation");
77
78 // Get tensor references
79 auto dy = attributes.get_dy();
80 auto x = attributes.get_x();
81 auto dx = attributes.get_dx();
82 auto scale = attributes.get_scale();
83 auto dscale = attributes.get_dscale();
84 auto dbias = attributes.get_dbias();
85
86 // SECTION 2: Validate Required Tensor Dimensions
87 // Why: All required tensors (x, dy, scale) must have dimensions set by user.
88 // Outputs (dx, dscale, dbias) are inferred, so not validated here.
89 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(x, 2, "Input tensor (x)"));
90 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(dy, 2, "Gradient input tensor (dy)"));
91 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(scale, 2, "Scale tensor"));
92
93 // SECTION 3: Validate Tensor Shape Consistency
94 // Why: Gradients flow through the same computational graph as forward pass.
95 // - dy has same shape as y (and therefore x) from forward pass
96 // - dx has same shape as x (gradient w.r.t. input)
97 // All gradient tensors must match the data tensor shapes they correspond to.
98
99 // Both x and dy validated in SECTION 2, can call directly
100 HIPDNN_CHECK_ERROR(
101 validateTensorShapesMatch(x, dy, "Input tensor (x)", "Gradient input tensor (dy)"));
102
103 // dx may not have dimensions set yet (will be inferred)
104 HIPDNN_CHECK_ERROR(validateTensorShapesMatchIfSet(
105 x, dx, "Input tensor (x)", "Gradient output tensor (dx)"));
106
107 // SECTION 4: Validate Channel Dimensions and Parameter Tensor Shapes
108 // Why: Parameter gradients (dscale, dbias) are accumulated per-channel over (N,H,W):
109 // dscale_c = Σ_{n,h,w} xhat[n,c,h,w] * dy[n,c,h,w] -> shape [1, C, 1, 1]
110 // dbias_c = Σ_{n,h,w} dy[n,c,h,w] -> shape [1, C, 1, 1]
111 // scale from forward pass is also per-channel [1, C, 1, 1].
112 // Saved statistics (mean_c, invStd_c) are per-channel for backward computation.
113
114 // Extract optional tensors once for reuse in both shape validation and consistency check
115 auto mean = attributes.get_mean();
116 auto invVar = attributes.get_inv_variance();
117
118 // Extract channel count - safe to access xDims[1] after SECTION 2 validation
119 auto& xDims = x->get_dim();
120 int64_t channels = xDims[1];
121
122 // Validate scale has correct channel-only shape (required user parameter)
123 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(scale, channels, "Scale tensor"));
124
125 // Validate gradient outputs (only if dimensions set, will be inferred otherwise)
126 HIPDNN_CHECK_ERROR(
127 validateChannelOnlyShapeIfSet(dscale, channels, "Scale gradient tensor (dscale)"));
128 HIPDNN_CHECK_ERROR(
129 validateChannelOnlyShapeIfSet(dbias, channels, "Bias gradient tensor (dbias)"));
130
131 // Validate optional saved statistics from forward pass (only if dimensions set)
132 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(mean, channels, "Mean tensor"));
133 HIPDNN_CHECK_ERROR(
134 validateChannelOnlyShapeIfSet(invVar, channels, "Inverse variance tensor"));
135
136 // SECTION 5: Validate Mean and Inverse Variance Consistency
137 // Why: Backward computation uses saved statistics (mean_c, invStd_c) from forward pass.
138 // These must be provided together (both or neither). If neither is provided, they will
139 // be recomputed during backward pass (less efficient but valid).
140 bool hasMean = (mean != nullptr);
141 bool hasInvVariance = (invVar != nullptr);
142 if(hasMean != hasInvVariance)
143 {
144 return {ErrorCode::INVALID_VALUE,
145 "BatchnormBackwardNode requires both mean and inv_variance to be set, or "
146 "neither"};
147 }
148
149 // SECTION 6: Validate Spatial Mode Constraints
150 HIPDNN_CHECK_ERROR(
151 validateBatchNormTrainingSpatialDimensions(x, scale, "Batch normalization backward"));
152
153 return {ErrorCode::OK, ""};
154 }

156 Error infer_properties_node() override
157 {
158 auto x = attributes.get_x();
159 auto dx = attributes.get_dx();
160 auto dscale = attributes.get_dscale();
161 auto dbias = attributes.get_dbias();
162
163 if(!x)
164 {
165 return {ErrorCode::ATTRIBUTE_NOT_SET,
166 "BatchnormBackwardNode missing x for setting properties"};
167 }
168 if(!dx)
169 {
170 return {ErrorCode::ATTRIBUTE_NOT_SET,
171 "BatchnormBackwardNode missing dx for setting properties"};
172 }
173 if(!dscale)
174 {
175 return {ErrorCode::ATTRIBUTE_NOT_SET,
176 "BatchnormBackwardNode missing dscale for setting properties"};
177 }
178 if(!dbias)
179 {
180 return {ErrorCode::ATTRIBUTE_NOT_SET,
181 "BatchnormBackwardNode missing dbias for setting properties"};
182 }
183
184 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
185
186 if(dx->get_dim().empty())
187 {
188 dx->set_dim(x->get_dim());
189 }
190
191 if(dx->get_stride().empty())
192 {
193 dx->set_stride(x->get_stride());
194 }
195
196 auto inferCTensor = [&](std::shared_ptr<TensorAttributes>& tensorToInfer) {
197 if(tensorToInfer->get_dim().empty())
198 {
199 std::vector<int64_t> tensorDims(x->get_dim().size(), 1);
200 tensorDims[1] = x->get_dim()[1];
201 tensorToInfer->set_dim(tensorDims);
202 }
203
204 if(tensorToInfer->get_stride().empty())
205 {
206 auto strideOrder = hipdnn_data_sdk::utilities::extractStrideOrder(x->get_stride());
207 tensorToInfer->set_stride(hipdnn_data_sdk::utilities::generateStrides(
208 tensorToInfer->get_dim(), strideOrder));
209 }
210 };
211
212 inferCTensor(dscale);
213 inferCTensor(dbias);
214
215 return {};
216 }
