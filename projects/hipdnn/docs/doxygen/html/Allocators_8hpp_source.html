120 template <typename U>
121 HostAllocator([[maybe_unused]] const HostAllocator<U>& other) noexcept
122 {
123 }
124
125 ~HostAllocator() override = default;
126
127 [[nodiscard]] T* allocate(std::size_t n) override
128 {
129 if(n > std::numeric_limits<std::size_t>::max() / sizeof(T))
130 {
131 throw std::bad_alloc();
132 }
133
134 auto ptr = static_cast<T*>(std::malloc(n * sizeof(T)));
135 if(!ptr)
136 {
137 throw std::bad_alloc();
138 }
139 return ptr;
140 }
141
142 void deallocate(T* p, [[maybe_unused]] std::size_t n) noexcept override
143 {
144 std::free(p);
145 }
