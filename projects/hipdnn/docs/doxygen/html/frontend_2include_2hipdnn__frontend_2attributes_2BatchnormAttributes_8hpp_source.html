311 flatbuffers::Offset<hipdnn_data_sdk::data_objects::BatchnormAttributes>
312 pack_attributes(flatbuffers::FlatBufferBuilder& builder) const // NOLINT
313 {
314 auto peerStatsVector = std::vector<int64_t>{};
315 for(const auto& peerStat : peer_stats)
316 {
317 if(peerStat)
318 {
319 peerStatsVector.emplace_back(peerStat->get_uid());
320 }
321 }
322
323 auto prevRunningMean = get_prev_running_mean();
324 auto prevRunningVariance = get_prev_running_variance();
325 auto momentum = get_momentum();
326 auto mean = get_mean();
327 auto invVariance = get_inv_variance();
328 auto nextRunningMean = get_next_running_mean();
329 auto nextRunningVariance = get_next_running_variance();
330
331 return hipdnn_data_sdk::data_objects::CreateBatchnormAttributesDirect(
332 builder,
333 get_x()->get_uid(),
334 get_scale()->get_uid(),
335 get_bias()->get_uid(),
336 get_epsilon()->get_uid(),
337 &peerStatsVector,
338 prevRunningMean ? flatbuffers::Optional<int64_t>(prevRunningMean->get_uid())
339 : flatbuffers::nullopt,
340 prevRunningVariance ? flatbuffers::Optional<int64_t>(prevRunningVariance->get_uid())
341 : flatbuffers::nullopt,
342 momentum ? flatbuffers::Optional<int64_t>(momentum->get_uid()) : flatbuffers::nullopt,
343 get_y()->get_uid(),
344 mean ? flatbuffers::Optional<int64_t>(mean->get_uid()) : flatbuffers::nullopt,
345 invVariance ? flatbuffers::Optional<int64_t>(invVariance->get_uid())
346 : flatbuffers::nullopt,
347 nextRunningMean ? flatbuffers::Optional<int64_t>(nextRunningMean->get_uid())
348 : flatbuffers::nullopt,
349 nextRunningVariance ? flatbuffers::Optional<int64_t>(nextRunningVariance->get_uid())
350 : flatbuffers::nullopt);
351 }

353 static BatchnormAttributes fromFlatBuffer(
354 const hipdnn_data_sdk::data_objects::BatchnormAttributes* fb,
355 const std::unordered_map<int64_t, std::shared_ptr<TensorAttributes>>& tensorMap)
356 {
357 BatchnormAttributes attr;
358
359 attr.set_x(tensorMap.at(fb->x_tensor_uid()));
360 attr.set_scale(tensorMap.at(fb->scale_tensor_uid()));
361 attr.set_bias(tensorMap.at(fb->bias_tensor_uid()));
362 attr.set_epsilon(tensorMap.at(fb->epsilon_tensor_uid()));
363
364 std::vector<std::shared_ptr<TensorAttributes>> peerStats;
365 if(fb->peer_stats_tensor_uid() != nullptr)
366 {
367 for(auto uid : *fb->peer_stats_tensor_uid())
368 {
369 peerStats.push_back(tensorMap.at(uid));
370 }
371 }
372 attr.set_peer_stats(peerStats);
373
374 if(fb->prev_running_mean_tensor_uid().has_value())
375 {
376 attr.set_prev_running_mean(tensorMap.at(fb->prev_running_mean_tensor_uid().value()));
377 }
378 if(fb->prev_running_variance_tensor_uid().has_value())
379 {
380 attr.set_prev_running_variance(
381 tensorMap.at(fb->prev_running_variance_tensor_uid().value()));
382 }
383 if(fb->momentum_tensor_uid().has_value())
384 {
385 attr.set_momentum(tensorMap.at(fb->momentum_tensor_uid().value()));
386 }
387
388 attr.set_y(tensorMap.at(fb->y_tensor_uid()));
389
390 if(fb->mean_tensor_uid().has_value())
391 {
392 attr.set_mean(tensorMap.at(fb->mean_tensor_uid().value()));
393 }
394 if(fb->inv_variance_tensor_uid().has_value())
395 {
396 attr.set_inv_variance(tensorMap.at(fb->inv_variance_tensor_uid().value()));
397 }
398 if(fb->next_running_mean_tensor_uid().has_value())
399 {
400 attr.set_next_running_mean(tensorMap.at(fb->next_running_mean_tensor_uid().value()));
401 }
402 if(fb->next_running_variance_tensor_uid().has_value())
403 {
404 attr.set_next_running_variance(
405 tensorMap.at(fb->next_running_variance_tensor_uid().value()));
406 }
407
408 return attr;
409 }
