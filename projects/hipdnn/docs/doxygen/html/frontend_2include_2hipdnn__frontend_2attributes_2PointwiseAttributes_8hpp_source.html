277 static PointwiseAttributes fromFlatBuffer(

280 {

292 }
293 if(fb->relu_lower_clip_slope().has_value())

296 }
297 if(fb->swish_beta().has_value())
298 {
299 attr.set_swish_beta(fb->swish_beta().value());
300 }
301 if(fb->elu_alpha().has_value())
302 {
303 attr.set_elu_alpha(fb->elu_alpha().value());
304 }
305 if(fb->softplus_beta().has_value())
306 {
307 attr.set_softplus_beta(fb->softplus_beta().value());
308 }
309 if(fb->axis_tensor_uid().has_value())
310 {
311 attr.set_axis(fb->axis_tensor_uid().value());
312 }
313
314 attr.set_input_0(tensorMap.at(fb->in_0_tensor_uid()));
315 if(fb->in_1_tensor_uid().has_value())
316 {
317 attr.set_input_1(tensorMap.at(fb->in_1_tensor_uid().value()));
318 }
319 if(fb->in_2_tensor_uid().has_value())
320 {
321 attr.set_input_2(tensorMap.at(fb->in_2_tensor_uid().value()));
322 }
323
324 attr.set_output_0(tensorMap.at(fb->out_0_tensor_uid()));
