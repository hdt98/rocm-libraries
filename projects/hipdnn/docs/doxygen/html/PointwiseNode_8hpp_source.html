52 Error infer_properties_node() override
53 {
54 auto out = attributes.get_output_0();
55 if(!out)
56 {
57 return {ErrorCode::ATTRIBUTE_NOT_SET,
58 "PointwiseNode missing output for setting properties"};
59 }
60
61 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
62
63 if(out->get_dim().empty())
64 {
65 std::vector<std::vector<int64_t>> inputShapes;
66
67 for(const auto& [_, tensor] : attributes.inputs)
68 {
69 if(tensor)
70 {
71 inputShapes.push_back(tensor->get_dim());
72 }
73 else
74 {
75 return {ErrorCode::INVALID_VALUE, "PointwiseNode has null input tensor"};
76 }
77 }
78
79 auto outputDims = out->get_dim();
80 HIPDNN_CHECK_ERROR(findCommonShape(inputShapes, outputDims));
81 out->set_dim(outputDims);
82 }
83
84 if(out->get_stride().empty())
85 {
86 for(const auto& [_, tensor] : attributes.inputs)
87 {
88 if(!tensor)

90 return {ErrorCode::INVALID_VALUE, "PointwiseNode has null input tensor"};
91 }
92
93 if(tensor->get_dim() == out->get_dim())
94 {
95 HIPDNN_FE_LOG_INFO(
96 "PointwiseNode {} inferring stride from input tensor {} for output {}",
97 attributes.get_name(),

102 }
103 }

126 Error checkInputsMatchOperationType() const
127 {
128 switch(attributes.inputs.size())
129 {
130 case 0:
131 return {ErrorCode::INVALID_VALUE, "PointwiseNode must have at least one input"};
132 case 1:
133 if(!isUnaryPointwiseMode(attributes.get_mode()))
134 {
135 return {ErrorCode::INVALID_VALUE,
136 "PointwiseNode with one input must have a unary operation"};
137 }
138 break;
139 case 2:
140 if(!isBinaryPointwiseMode(attributes.get_mode()))
141 {
142 return {ErrorCode::INVALID_VALUE,
143 "PointwiseNode with two inputs must have a binary operation"};
144 }
145 break;
146 case 3:
147 if(!isTernaryPointwiseMode(attributes.get_mode()))
148 {
149 return {ErrorCode::INVALID_VALUE,
150 "PointwiseNode with three inputs must have a ternary operation"};
151 }
152 break;
153 default:
154 return {ErrorCode::INVALID_VALUE,
155 "PointwiseNode can only have one, two, or three inputs"};
156 }
157
158 return {};
159 }
160
161 Error checkInputsAndOutputsAreBroadcastCompatible() const
162 {
163 auto output = attributes.get_output_0();
164 if(output && !output->get_dim().empty())
165 {
166 const auto& outputDims = output->get_dim();
167
168 for(const auto& [_, inputTensor] : attributes.inputs)
169 {
170 if(inputTensor)
171 {
172 const auto& inputDims = inputTensor->get_dim();
173
174 if(!hipdnn_data_sdk::utilities::areDimensionsBroadcastCompatible(inputDims,
175 outputDims))
176 {
177 return {ErrorCode::INVALID_VALUE,
178 "PointwiseNode input '" + inputTensor->get_name()
179 + "' has dimensions incompatible with output. "
180 + "All inputs must be broadcastable to output dimensions."};
181 }
182 }
183 else
184 {
185 return {ErrorCode::INVALID_VALUE, "PointwiseNode has null input tensor"};
186 }
187 }
188 }
189
190 return {};
191 }
192};
