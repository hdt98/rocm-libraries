122// Any class extending BaseNode must have an attributes member with an inputs & outputs map.
123// The map needs to have TensorAttributes as the value.
124// BaseNode uses this to gather tensor uids, and populate unset ones.
125template <typename DerivedT>
126class BaseNode : public INode
127{
128private:
129 DerivedT& self()
130 {
131 return static_cast<DerivedT&>(*this);
132 }
133 const DerivedT& self() const
134 {
135 return static_cast<const DerivedT&>(*this);
136 }
137
138public:
139 // NOLINTNEXTLINE(readability-identifier-naming)
140 void gather_hipdnn_tensors(
141 std::unordered_set<std::shared_ptr<TensorAttributes>>& allTensors) const override
142 {
143 for(auto& [_, tensor] : self().attributes.inputs)
144 {
145 if(tensor)
146 {
147 allTensors.insert(tensor);
148 }
149 }
150
151 for(auto& [_, tensor] : self().attributes.outputs)
152 {
153 if(tensor)
154 {
155 allTensors.insert(tensor);
156 }
157 }
158 }
159
160 Error post_validate_node() const override // NOLINT(readability-identifier-naming)
161 {
162 if(self().attributes.compute_data_type == DataType::NOT_SET)
163 {
164 return {ErrorCode::ATTRIBUTE_NOT_SET,
165 "Node " + self().attributes.name + " does not have a compute_data_type set"};
166 }
167
168 for(const auto& tensorAttr : getNodeOutputTensorAttributes())
169 {
170 HIPDNN_CHECK_ERROR(tensorAttr->validate());
171 }
172
173 return {ErrorCode::OK, ""};
174 }
175
176 std::vector<std::shared_ptr<TensorAttributes>> getNodeInputTensorAttributes() const override
177 {
178 std::vector<std::shared_ptr<TensorAttributes>> inputAttributes;
179 for(auto& tensorAttrPair : self().attributes.inputs)
180 {
181 if(tensorAttrPair.second)
182 {
183 inputAttributes.push_back(tensorAttrPair.second);
184 }
185 }
186
187 return inputAttributes;
188 }
189
190 std::vector<std::shared_ptr<TensorAttributes>> getNodeOutputTensorAttributes() const override
191 {
192 std::vector<std::shared_ptr<TensorAttributes>> outputAttributes;
193 for(auto& tensorAttrPair : self().attributes.outputs)
194 {
195 if(tensorAttrPair.second)
196 {
197 outputAttributes.push_back(tensorAttrPair.second);
198 }
199 }
200
201 return outputAttributes;
202 }
203
204protected:
205 using INode::INode;
206};
