26 Error pre_validate_node() const override
27 {
28 // ====================================================================
29 // BATCH NORMALIZATION FORWARD TRAINING VALIDATION
30 // (Spatial Mode: per-channel statistics over N×H×W for 4D, N×D×H×W for 5D)
31 // ====================================================================
32 // Algorithm Overview:
33 // For each channel c, BN computes batch statistics over (N,H,W):
34 // mean_c = (1/m) * Σ_{n,h,w} x[n,c,h,w] where m = N*H*W
35 // var_c = (1/m) * Σ_{n,h,w} (x[n,c,h,w] - mean_c)²
36 //
37 // Normalizes: xhat[n,c,h,w] = (x[n,c,h,w] - mean_c) / sqrt(var_c + ε)
38 // Transforms: y[n,c,h,w] = scale_c * xhat[n,c,h,w] + bias_c
39 //
40 // Optionally outputs mean_c and invStd_c to device buffers
41 // (consumed by backward pass for gradient computation)
42 //
43 // Updates running stats: runMean_c = (1-momentum)*runMean_c + momentum*mean_c
44 // runVar_c = (1-momentum)*runVar_c + momentum*var_c
45 // ====================================================================
46
47 // SECTION 1: Validate Required Tensor Pointers
48 HIPDNN_RETURN_IF_FALSE(attributes.get_x(),
49 ErrorCode::ATTRIBUTE_NOT_SET,
50 "BatchnormNode missing x for pre-validation");
51
52 HIPDNN_RETURN_IF_FALSE(attributes.get_scale(),
53 ErrorCode::ATTRIBUTE_NOT_SET,
54 "BatchnormNode missing scale for pre-validation");
55
56 HIPDNN_RETURN_IF_FALSE(attributes.get_bias(),
57 ErrorCode::ATTRIBUTE_NOT_SET,
58 "BatchnormNode missing bias for pre-validation");
59
60 HIPDNN_RETURN_IF_FALSE(attributes.get_y(),
61 ErrorCode::ATTRIBUTE_NOT_SET,
62 "BatchnormNode missing y for pre-validation");
63
64 HIPDNN_RETURN_IF_FALSE(attributes.get_epsilon(),
65 ErrorCode::ATTRIBUTE_NOT_SET,
66 "BatchnormNode missing epsilon for pre-validation");
67
68 // Get tensor references
69 auto x = attributes.get_x();
70 auto y = attributes.get_y();
71 auto scale = attributes.get_scale();
72 auto bias = attributes.get_bias();
73 auto epsilon = attributes.get_epsilon();
74
75 // SECTION 2: Validate Required Parameter Dimensions
76 // Why: All required parameters (x, scale, bias, epsilon) must have dimensions
77 // set by user. Validate them upfront before proceeding with shape checks.
78 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(x, 2, "Input tensor (x)"));
79 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(scale, 2, "Scale tensor"));
80 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(bias, 2, "Bias tensor"));
81
82 // Epsilon (ε) provides numerical stability: xhat = (x - mean) / sqrt(var + ε)
83 // Without ε, division by zero occurs when var ≈ 0. Must be a scalar.
84 HIPDNN_CHECK_ERROR(validateScalarParameter(epsilon, "Epsilon"));
85
86 // SECTION 3: Validate Output Tensor Shape Consistency
87 // Why: BN preserves tensor shape - it only transforms values, not dimensions.
88 // Output y[n,c,h,w] has same shape as input x[n,c,h,w].
89 HIPDNN_CHECK_ERROR(
90 validateTensorShapesMatchIfSet(x, y, "Input tensor (x)", "Output tensor (y)"));
91
92 // SECTION 4: Validate Channel Dimensions and Parameter Tensor Shapes
93 // Why: All BN parameters (scale, bias, mean, variance) are per-channel with
94 // shape [1, C, 1, 1, ...]. This is because:
95 // - Each channel c has its own statistics: mean_c, var_c
96 // - Each channel c has its own learnable parameters: scale_c, bias_c
97 // - scale_c controls feature importance/gain after normalization
98 // - bias_c controls activation threshold (e.g., for ReLU: active when scale_c*xhat + bias_c > 0)
99
100 // Extract channel count - safe to access xDims[1] after SECTION 2 validation
101 auto& xDims = x->get_dim();
102 int64_t channels = xDims[1];
103
104 // Validate scale has correct channel-only shape (required user parameter)
105 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(scale, channels, "Scale tensor"));
106
107 // Validate bias has correct channel-only shape (required user parameter)
108 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(bias, channels, "Bias tensor"));
109
110 // Validate optional mean and inv_variance tensors (only if dimensions set)
111 HIPDNN_CHECK_ERROR(
112 validateChannelOnlyShapeIfSet(attributes.get_mean(), channels, "Mean tensor"));
113 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(
114 attributes.get_inv_variance(), channels, "Inverse variance tensor"));
115
116 // SECTION 5: Validate Running Stats Consistency
117 // Why: Running statistics are updated together during training:
118 // nextRunMean_c = (1-momentum)*prevRunMean_c + momentum*batchMean_c
119 // nextRunVar_c = (1-momentum)*prevRunVar_c + momentum*batchVar_c
120 // These are used for inference after training completes. If any are provided,
121 // all must be provided to ensure consistent exponential moving average updates.
122 auto prevRunningMean = attributes.get_prev_running_mean();
123 auto prevRunningVar = attributes.get_prev_running_variance();
124 auto nextRunningMean = attributes.get_next_running_mean();
125 auto nextRunningVar = attributes.get_next_running_variance();
126
127 // If any running stat is provided, all must be provided
128 bool hasPrevRunningMean = prevRunningMean != nullptr;
129 bool hasPrevRunningVar = prevRunningVar != nullptr;
130 bool hasNextRunningMean = nextRunningMean != nullptr;
131 bool hasNextRunningVar = nextRunningVar != nullptr;
132
133 if(hasPrevRunningMean || hasPrevRunningVar || hasNextRunningMean || hasNextRunningVar)
134 {
135 HIPDNN_RETURN_IF_FALSE(
136 hasPrevRunningMean && hasPrevRunningVar && hasNextRunningMean && hasNextRunningVar,
137 ErrorCode::INVALID_VALUE,
138 "BatchnormNode: If any running statistics are provided, all running "
139 "statistics "
140 "(prev_running_mean, prev_running_variance, next_running_mean, "
141 "next_running_variance) must be provided");
142
143 // Validate running stats have correct shapes (only if dimensions set)
144 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(
145 prevRunningMean, channels, "Previous running mean tensor"));
146 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(
147 prevRunningVar, channels, "Previous running variance tensor"));
148 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(
149 nextRunningMean, channels, "Next running mean tensor"));
150 HIPDNN_CHECK_ERROR(validateChannelOnlyShapeIfSet(
151 nextRunningVar, channels, "Next running variance tensor"));
152 }
153
154 // SECTION 6: Validate Spatial Mode Constraints
155 // Why: For spatial BN, statistics are computed over N*H*W elements per channel.
156 // We need N*H*W > 1 to compute meaningful statistics (mean and variance).
157 // With only 1 element, variance is undefined and normalization degenerates.
158 HIPDNN_CHECK_ERROR(
159 validateBatchNormTrainingSpatialDimensions(x, scale, "Batch normalization training"));
160
161 return {ErrorCode::OK, ""};
162 }

164 Error infer_properties_node() override
165 {
166 auto x = attributes.get_x();
167 auto y = attributes.get_y();
168
169 if(!x)
170 {
171 return {ErrorCode::ATTRIBUTE_NOT_SET, "BatchnormNode missing x for setting properties"};
172 }
173
174 if(!y)
175 {
176 return {ErrorCode::ATTRIBUTE_NOT_SET, "BatchnormNode missing y for setting properties"};
177 }
178
179 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
180
181 if(y->get_dim().empty())
182 {
183 y->set_dim(x->get_dim());
184 }
185
186 if(y->get_stride().empty())
187 {
188 y->set_stride(x->get_stride());
189 }
190
191 auto inferCTensor = [&](std::shared_ptr<TensorAttributes>& tensorToInfer) {
192 if(tensorToInfer->get_dim().empty())
193 {
194 std::vector<int64_t> tensorDims(x->get_dim().size(), 1);
195 tensorDims[1] = x->get_dim()[1];
196 tensorToInfer->set_dim(tensorDims);
197 }
198
199 if(tensorToInfer->get_stride().empty())
200 {
201 auto strideOrder = hipdnn_data_sdk::utilities::extractStrideOrder(x->get_stride());
202 tensorToInfer->set_stride(hipdnn_data_sdk::utilities::generateStrides(
203 tensorToInfer->get_dim(), strideOrder));
204 }
205 };
206
207 auto mean = attributes.get_mean();
208 auto invVar = attributes.get_inv_variance();
209
210 if(mean)
211 {
212 inferCTensor(mean);
213 }
214
215 if(invVar)
216 {
217 inferCTensor(invVar);
218 }
219
220 auto prevRunningMean = attributes.get_prev_running_mean();
221 auto prevRunningVar = attributes.get_prev_running_variance();
222
223 auto nextRunningMean = attributes.get_next_running_mean();
224 auto nextRunningVar = attributes.get_next_running_variance();
225
226 if(prevRunningMean && prevRunningVar && nextRunningMean && nextRunningVar)
227 {
228 inferCTensor(nextRunningMean);
229 inferCTensor(nextRunningVar);
230 }
231
232 return {};
233 }
