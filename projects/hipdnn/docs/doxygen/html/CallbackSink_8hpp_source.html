19inline hipdnnSeverity_t spdlogToHipdnnSeverity(spdlog::level::level_enum level)
20{
21 switch(level)
22 {
23 case spdlog::level::critical:
24 return HIPDNN_SEV_FATAL;
25 case spdlog::level::err:
26 return HIPDNN_SEV_ERROR;
27 case spdlog::level::warn:
28 return HIPDNN_SEV_WARN;
29 case spdlog::level::off:
30 return HIPDNN_SEV_OFF;
31 default:
32 return HIPDNN_SEV_INFO;
33 }
34}

36// This warning is for cross-compiler compatibility, so it's safe to ignore since we only use Clang.
37// Moreover, MSVC is likely the incompatible compiler here.
38// NOLINTBEGIN(portability-template-virtual-member-function)
39template <typename Mutex>
40class CallbackSink final : public spdlog::sinks::base_sink<Mutex>
41{
42public:
43 explicit CallbackSink(hipdnnCallback_t callback)
44 : _callbackFn{callback}
45 {
46 }
47 ~CallbackSink() override = default;
48
49protected:
50 void sink_it_(const spdlog::details::log_msg& msg) override
51 {
52 if(_callbackFn == nullptr)
53 {
54 return;
55 }
56
57 spdlog::memory_buf_t formatted;
58 spdlog::sinks::base_sink<Mutex>::formatter_->format(msg, formatted);
59 std::string formattedStr(formatted.data(), formatted.size());
60
61 while(!formattedStr.empty() && (formattedStr.back() == '\n' || formattedStr.back() == '\r'))
62 {
63 formattedStr.pop_back();
64 }
65
66 hipdnnSeverity_t severity = spdlogToHipdnnSeverity(msg.level);
67
68 _callbackFn(severity, formattedStr.c_str());
69 }
70
71 void flush_() override {}
72
73private:
74 hipdnnCallback_t _callbackFn;
75};
76// NOLINTEND(portability-template-virtual-member-function)

80inline std::shared_ptr<spdlog::logger> createAsyncCallbackLoggerMt(hipdnnCallback_t callback,
81 const std::string& source)
82{
83 auto sink = std::make_shared<hipdnn_data_sdk::logging::CallbackSinkMt>(callback);
84 auto logger = std::make_shared<spdlog::async_logger>(
85 source, sink, spdlog::thread_pool(), spdlog::async_overflow_policy::block);
86 logger->set_pattern(generatePatternString(source));
87 logger->flush_on(spdlog::level::info);
88 return logger;
89}
