18// When an error occurs, get the backend error string and append it to the error_message.
19#define HIPDNN_RETURN_ON_BACKEND_FAILURE(backend_status, error_message) \
20 do \
21 { \
22 if((backend_status) != HIPDNN_STATUS_SUCCESS) \
23 { \
24 std::array<char, 1024> backend_err_msg{}; \
25 hipdnn_frontend::detail::hipdnnBackend()->getLastErrorString(backend_err_msg.data(), \
26 backend_err_msg.size()); \
27 std::string full_error_msg \
28 = std::string(error_message) + " Backend error: " + backend_err_msg.data(); \
29 return Error(ErrorCode::HIPDNN_BACKEND_ERROR, full_error_msg); \
30 } \
31 } while(0)
