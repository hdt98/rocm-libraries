16/*
17 * The IGraph interface expects that any implementations have the graph sorted in topological order.
18 * The graph must also have no cycles and be fully connected(no orphan nodes). We also expect
19 * that the all tensors in the graph have unique uids.
20*/
21class IGraph
22{
23public:
24 virtual ~IGraph() = default;
25
26 virtual const hipdnn_data_sdk::data_objects::Graph& getGraph() const = 0;
27 virtual bool isValid() const = 0;
28 virtual uint32_t nodeCount() const = 0;
29 virtual bool hasOnlySupportedAttributes(
30 std::set<hipdnn_data_sdk::data_objects::NodeAttributes> supportedAttributes) const
31 = 0;
32 virtual const hipdnn_data_sdk::data_objects::Node& getNode(uint32_t index) const = 0;
33 virtual const INodeWrapper& getNodeWrapper(uint32_t index) const = 0;
34 virtual const std::vector<std::unique_ptr<INodeWrapper>>& nodeWrappers() const = 0;
35 virtual const std::unordered_map<int64_t,
36 const hipdnn_data_sdk::data_objects::TensorAttributes*>&
37 getTensorMap() const
38 = 0;
39};

41class GraphWrapper : public IGraph
42{
43public:
44 explicit GraphWrapper(const void* buffer, size_t size)
45 {
46 if(buffer != nullptr)
47 {
48 flatbuffers::Verifier verifier(static_cast<const uint8_t*>(buffer), size);
49 if(verifier.VerifyBuffer<hipdnn_data_sdk::data_objects::Graph>())
50 {
51 _shallowGraph = flatbuffers::GetRoot<hipdnn_data_sdk::data_objects::Graph>(buffer);
52 }
53 }
54 }
55
56 const hipdnn_data_sdk::data_objects::Graph& getGraph() const override
57 {
58 throwIfNotValid();
59 return *_shallowGraph;
60 }
61
62 bool isValid() const override
63 {
64 return _shallowGraph != nullptr;
65 }
66
67 uint32_t nodeCount() const override
68 {
69 throwIfNotValid();
70 auto nodes = _shallowGraph->nodes();
71 if(nodes == nullptr)
72 {
73 return 0;
74 }
75 return static_cast<uint32_t>(nodes->size());
76 }
77
78 bool hasOnlySupportedAttributes(
79 std::set<hipdnn_data_sdk::data_objects::NodeAttributes> supportedAttributes) const override
80 {
81 throwIfNotValid();
82
83 auto nodes = _shallowGraph->nodes();
84 if(nodes == nullptr)
85 {
86 return true; // No nodes means no unsupported attributes
87 }
88
89 return std::all_of(nodes->begin(), nodes->end(), [&](const auto node) {
90 return supportedAttributes.find(node->attributes_type()) != supportedAttributes.end();
91 });
92 }
93
94 const hipdnn_data_sdk::data_objects::Node& getNode(uint32_t index) const override
95 {
96 throwIfNotValid();
97
98 auto nodes = _shallowGraph->nodes();
99 if(nodes == nullptr)
100 {
101 throw std::out_of_range("No nodes in graph");
102 }
103
104 if(index >= nodes->size())
105 {
106 throw std::out_of_range("Index out of range for graph nodes");
107 }
108
109 return *nodes->Get(index);
110 }
111
112 const INodeWrapper& getNodeWrapper(uint32_t index) const override
113 {
114 throwIfNotValid();
115
116 lazyInitNodeWrappers();
117
118 if(index >= _nodeWrappers.size())
119 {
120 throw std::out_of_range("Index out of range for graph nodes");
121 }
122 return *_nodeWrappers[index];
123 }
124
125 const std::vector<std::unique_ptr<INodeWrapper>>& nodeWrappers() const override
126 {
127 lazyInitNodeWrappers();
128
129 return _nodeWrappers;
130 }
131
132 const std::unordered_map<int64_t, const hipdnn_data_sdk::data_objects::TensorAttributes*>&
133 getTensorMap() const override
134 {
135 throwIfNotValid();
136
137 if(!_tensorMap.empty())
138 {
139 return _tensorMap;
140 }
141
142 auto tensors = _shallowGraph->tensors();
143 if(tensors != nullptr)
144 {
145 for(const auto tensor : *tensors)
146 {
147 _tensorMap[tensor->uid()] = tensor;
148 }
149 }
150
151 return _tensorMap;
152 }
153
154private:
155 void throwIfNotValid() const
156 {
157 if(!isValid())
158 {
159 throw std::invalid_argument("Graph is not valid");
160 }
161 }
162
163 void lazyInitNodeWrappers() const
164 {
165 if(_nodeWrappers.empty())
166 {
167 auto nodes = _shallowGraph->nodes();
168 if(nodes == nullptr)
169 {
170 throw std::out_of_range("No nodes in graph");
171 }
172
173 _nodeWrappers.reserve(nodes->size());
174 for(const auto node : *nodes)
175 {
176 _nodeWrappers.push_back(std::make_unique<NodeWrapper>(node));
177 }
178 }
179 }
180
181 // Pointer to the flatbuffer representation of the graph. We do not own this memory
182 // as were just reading from the buffer passed during construction.
183 const hipdnn_data_sdk::data_objects::Graph* _shallowGraph = nullptr;
184
185 //lazy init state;
186 mutable std::unordered_map<int64_t, const hipdnn_data_sdk::data_objects::TensorAttributes*>
187 _tensorMap;
188
189 mutable std::vector<std::unique_ptr<INodeWrapper>> _nodeWrappers;
190};
