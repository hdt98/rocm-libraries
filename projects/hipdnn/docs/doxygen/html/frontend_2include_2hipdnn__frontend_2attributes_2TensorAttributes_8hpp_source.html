342 Error validate() const
343 {
344 if(_dataType == DataType::NOT_SET)
345 {
346 return {ErrorCode::ATTRIBUTE_NOT_SET,
347 "Tensor " + _name + " does not have a data type set"};
348 }
349
350 HIPDNN_RETURN_IF_TRUE(_isVirtual && get_pass_by_value(),
351 ErrorCode::INVALID_VALUE,
352 "Tensor " + _name + " cannot be virtual and pass by value");
353 HIPDNN_RETURN_IF_NE(_dim.size(),
354 _stride.size(),
355 ErrorCode::INVALID_VALUE,
356 "Tensor " + _name + " dims and strides have different sizes");
357
358 HIPDNN_RETURN_IF_TRUE(_dim.empty(),
359 ErrorCode::ATTRIBUTE_NOT_SET,
360 "Tensor " + _name + " dims must be non-empty");
361
362 auto isPositive = [](int64_t value) constexpr { return value > 0; };
363 HIPDNN_RETURN_IF_FALSE(std::all_of(_dim.begin(), _dim.end(), isPositive),
364 ErrorCode::INVALID_VALUE,
365 "Tensor " + _name + " must have only positive dimensions");
366
367 return {ErrorCode::OK, ""};
368 }

370 flatbuffers::Offset<hipdnn_data_sdk::data_objects::TensorAttributes>
371 pack_attributes(flatbuffers::FlatBufferBuilder& builder) const // NOLINT
372 {
373 auto result = std::visit(
374 [&](auto&& arg) -> std::pair<hipdnn_data_sdk::data_objects::TensorValue,
375 flatbuffers::Offset<void>> {
376 using T = std::decay_t<decltype(arg)>;
377 if constexpr(std::is_same_v<T, float>)
378 {
379 hipdnn_data_sdk::data_objects::Float32Value floatVal(arg);
380 return {hipdnn_data_sdk::data_objects::TensorValue::Float32Value,
381 builder.CreateStruct(floatVal).Union()};
382 }
383 else if constexpr(std::is_same_v<T, double>)
384 {
385 hipdnn_data_sdk::data_objects::Float64Value doubleVal(arg);
386 return {hipdnn_data_sdk::data_objects::TensorValue::Float64Value,
387 builder.CreateStruct(doubleVal).Union()};
388 }
389 else if constexpr(std::is_same_v<T, half>)
390 {
391 hipdnn_data_sdk::data_objects::Float16Value halfVal(static_cast<float>(arg));
392 return {hipdnn_data_sdk::data_objects::TensorValue::Float16Value,
393 builder.CreateStruct(halfVal).Union()};
394 }
395 else if constexpr(std::is_same_v<T, bfloat16>)
396 {
397 hipdnn_data_sdk::data_objects::BFloat16Value bfVal(static_cast<float>(arg));
398 return {hipdnn_data_sdk::data_objects::TensorValue::BFloat16Value,
399 builder.CreateStruct(bfVal).Union()};
400 }
401 else if constexpr(std::is_same_v<T, uint8_t>)
402 {
403 hipdnn_data_sdk::data_objects::Float8Value uint8Val(arg);
404 return {hipdnn_data_sdk::data_objects::TensorValue::Float8Value,
405 builder.CreateStruct(uint8Val).Union()};
406 }
407 else if constexpr(std::is_same_v<T, int32_t>)
408 {
409 hipdnn_data_sdk::data_objects::Int32Value int32Val(arg);
410 return {hipdnn_data_sdk::data_objects::TensorValue::Int32Value,
411 builder.CreateStruct(int32Val).Union()};
412 }
413 else
414 {
415 // For std::monostate case
416 return {hipdnn_data_sdk::data_objects::TensorValue::NONE, 0};
417 }
418 },
419 _value);
420
421 return hipdnn_data_sdk::data_objects::CreateTensorAttributesDirect(builder,
422 _uid,
423 _name.c_str(),
424 toSdkType(_dataType),
425 &_stride,
426 &_dim,
427 _isVirtual,
428 result.first,
429 result.second);
430 }

432 static std::shared_ptr<TensorAttributes>
433 fromFlatBuffer(const hipdnn_data_sdk::data_objects::TensorAttributes* fb)
434 {
435 if(fb == nullptr)
436 {
437 return nullptr;
438 }
439
440 auto tensor = std::make_shared<TensorAttributes>();
441
442 tensor->set_uid(fb->uid());
443
444 if(fb->name() != nullptr)
445 {
446 tensor->set_name(fb->name()->c_str());
447 }
448
449 tensor->set_data_type(fromSdkType(fb->data_type()));
450
451 if(fb->dims() != nullptr)
452 {
453 std::vector<int64_t> dims(fb->dims()->begin(), fb->dims()->end());
454 tensor->set_dim(dims);
455 }
456
457 if(fb->strides() != nullptr)
458 {
459 std::vector<int64_t> strides(fb->strides()->begin(), fb->strides()->end());
460 tensor->set_stride(strides);
461 }
462
463 tensor->set_is_virtual(fb->virtual_());
464
465 auto valueType = fb->value_type();
466 if(valueType != hipdnn_data_sdk::data_objects::TensorValue::NONE)
467 {
468 switch(valueType)
469 {
470 case hipdnn_data_sdk::data_objects::TensorValue::Float32Value:
471 tensor->set_value(fb->value_as_Float32Value()->value());
472 break;
473 case hipdnn_data_sdk::data_objects::TensorValue::Float64Value:
474 tensor->set_value(fb->value_as_Float64Value()->value());
475 break;
476 case hipdnn_data_sdk::data_objects::TensorValue::Float16Value:
477 tensor->set_value(static_cast<half>(fb->value_as_Float16Value()->value()));
478 break;
479 case hipdnn_data_sdk::data_objects::TensorValue::BFloat16Value:
480 tensor->set_value(static_cast<bfloat16>(fb->value_as_BFloat16Value()->value()));
481 break;
482 case hipdnn_data_sdk::data_objects::TensorValue::Float8Value:
483 tensor->set_value(fb->value_as_Float8Value()->value());
484 break;
485 case hipdnn_data_sdk::data_objects::TensorValue::Int32Value:
486 tensor->set_value(fb->value_as_Int32Value()->value());
487 break;
488 default:
489 break;
490 }
491 }
492
493 return tensor;
494 }
