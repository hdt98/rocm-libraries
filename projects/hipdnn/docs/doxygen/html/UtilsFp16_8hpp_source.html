40inline __HOST_DEVICE__ half hmax(const half a, const half b)
41{
42 if(hisnan(a) && !hisnan(b))
43 {
44 return b;
45 }
46 if(!hisnan(a) && hisnan(b))
47 {
48 return a;
49 }
50 if(hisnan(a) && hisnan(b))
51 {
52 return HIPDNN_NAN_FP16;
53 }
54 if(static_cast<__half_raw>(a).x > static_cast<__half_raw>(b).x)
55 {
56 return __half_raw{static_cast<__half_raw>(a).x};
57 }
58 return __half_raw{static_cast<__half_raw>(b).x};
59}
