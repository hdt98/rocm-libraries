43inline __HOST_DEVICE__ hip_bfloat16 hmax(const hip_bfloat16 a, const hip_bfloat16 b)
44{
45 auto aNan = hisnan(a);
46 auto bNan = hisnan(b);
47
48 if(aNan || bNan)
49 {
50 if(aNan && bNan)
51 {
52 return HIPDNN_NAN_BF16; // return canonical NaN
53 }
54
55 return aNan ? b : a;
56 }
57
58 return a.data > b.data ? a : b;
59}
