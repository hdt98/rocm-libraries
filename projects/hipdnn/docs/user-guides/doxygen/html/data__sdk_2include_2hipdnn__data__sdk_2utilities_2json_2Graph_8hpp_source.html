93inline auto to<data_objects::Node>(flatbuffers::FlatBufferBuilder& builder,
94 const nlohmann::json& entry)
95{
96 auto type = entry.at("type").get<data_objects::NodeAttributes>();
97 auto name = entry.at("name").get<std::string>();
98 auto computeDataType = entry.at("compute_data_type").get<data_objects::DataType>();
99
100 flatbuffers::Offset<void> node = [&]() {
101 switch(type)
102 {
103 case data_objects::NodeAttributes::BatchnormInferenceAttributes:
104 return to<data_objects::BatchnormInferenceAttributes>(builder, entry).Union();
105 case data_objects::NodeAttributes::BatchnormInferenceAttributesVarianceExt:
106 return to<data_objects::BatchnormInferenceAttributesVarianceExt>(builder, entry)
107 .Union();
108 case data_objects::NodeAttributes::BatchnormBackwardAttributes:
109 return to<data_objects::BatchnormBackwardAttributes>(builder, entry).Union();
110 case data_objects::NodeAttributes::BatchnormAttributes:
111 return to<data_objects::BatchnormAttributes>(builder, entry).Union();
112 case data_objects::NodeAttributes::PointwiseAttributes:
113 return to<data_objects::PointwiseAttributes>(builder, entry).Union();
114 case data_objects::NodeAttributes::ConvolutionFwdAttributes:
115 return to<data_objects::ConvolutionFwdAttributes>(builder, entry).Union();
116 case data_objects::NodeAttributes::ConvolutionBwdAttributes:
117 return to<data_objects::ConvolutionBwdAttributes>(builder, entry).Union();
118 case data_objects::NodeAttributes::ConvolutionWrwAttributes:
119 return to<data_objects::ConvolutionWrwAttributes>(builder, entry).Union();
120 default:
121 throw std::runtime_error(
122 "hipdnn_data_sdk::json::to<data_objects::Node>(): Unsupported NodeAttributes type: "
123 + std::string{EnumNameNodeAttributes(type)});
124 }
125 }();
126
127 return data_objects::CreateNodeDirect(builder, name.c_str(), computeDataType, type, node);
128}
