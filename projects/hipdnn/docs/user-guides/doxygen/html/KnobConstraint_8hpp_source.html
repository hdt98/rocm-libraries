60class IntConstraint : public IConstraint
61{
62public:
70 IntConstraint(int64_t minValue,
71 int64_t maxValue,
72 int64_t step = 1,
73 std::unordered_set<int64_t> validValues = {})
74 : _minValue(minValue)
75 , _maxValue(maxValue)
76 , _step(step)
77 , _validValues(std::move(validValues))
78 {
79 }
80
81 Error validateKnobSetting(const KnobSetting& setting) const override
82 {
83 auto value = std::get_if<int64_t>(&setting.value());
84 if(value == nullptr)
85 {
86 return {ErrorCode::INVALID_VALUE, "KnobSetting does not contain an integer value"};
87 }
88
89 int64_t val = *value;
90
91 // If explicit valid values are specified, check against them
92 if(!_validValues.empty())
93 {
94 if(_validValues.count(val) == 0)
95 {
96 std::ostringstream oss;
97 oss << "Value " << val << " is not in the list of valid values: ";
98 std::vector<int64_t> sortedValues(_validValues.begin(), _validValues.end());
99 std::sort(sortedValues.begin(), sortedValues.end());
100 return {ErrorCode::INVALID_VALUE, oss.str()};
101 }
102 return {ErrorCode::OK, ""};
103 }
104
105 // Otherwise check min/max/step
106 if(val < _minValue || val > _maxValue)
107 {
108 std::ostringstream oss;
109 oss << "Value " << val << " is out of range [" << _minValue << ", " << _maxValue << "]";
110 return {ErrorCode::INVALID_VALUE, oss.str()};
111 }
112
113 if(_step > 1 && ((val - _minValue) % _step) != 0)
114 {
115 std::ostringstream oss;
116 oss << "Value " << val << " does not satisfy step constraint (step=" << _step
117 << ", min=" << _minValue << ")";
118 return {ErrorCode::INVALID_VALUE, oss.str()};
119 }
120
121 return {ErrorCode::OK, ""};
122 }
123
124 std::string toString() const override
125 {
126 std::ostringstream oss;
127 oss << "IntConstraint{min=" << _minValue << ", max=" << _maxValue << ", step=" << _step;
128 if(!_validValues.empty())
129 {
130 std::vector<int64_t> sortedValues(_validValues.begin(), _validValues.end());
131 std::sort(sortedValues.begin(), sortedValues.end());
132 oss << ", validValues=";
133 hipdnn_data_sdk::utilities::vecToStream(oss, sortedValues);
134 }
135 oss << "}";
136 return oss.str();
137 }
138
140 int64_t getMinValue() const
141 {
142 return _minValue;
143 }
144
146 int64_t getMaxValue() const
147 {
148 return _maxValue;
149 }
150
152 int64_t getStep() const
153 {
154 return _step;
155 }
156
158 const std::unordered_set<int64_t>& getValidValues() const
159 {
160 return _validValues;
161 }
162
163private:
164 int64_t _minValue;
165 int64_t _maxValue;
166 int64_t _step;
167 std::unordered_set<int64_t> _validValues;
168};
