73inline Config parseCommandLineArgs(int argc, char* argv[])
74{
75 auto config = Config{};
76
77 for(int i = 1; i < argc; ++i)
78 {
79 auto arg = std::string(argv[i]);
80
81 if(arg == "--verify-cpu" || arg == "-vc")
82 {
83 config.cpuValidation = true;
84 }
85 else if(arg == "--batch-stats-only")
86 {
87 config.useRunningStats = false;
88 }
89 else if(arg == "--full-training")
90 {
91 config.useRunningStats = true;
92 }
93 else if(arg == "--help" || arg == "-h")
94 {
95 printSampleHelp(argv[0]);
96 exit(EXIT_SUCCESS);
97 }
98 else
99 {
100 std::cerr << "Unknown argument: " << arg << std::endl;
101 printSampleHelp(argv[0]);
102 exit(EXIT_FAILURE);
103 }
104 }
105
106 return config;
107}

109template <typename F>
110bool run(F&& f)
111{
112 bool allPassed = true;
113 allPassed &= f.template operator()<float, float>(TensorLayout::NCHW);
114 allPassed &= f.template operator()<half, float>(TensorLayout::NCHW);
115 allPassed &= f.template operator()<hip_bfloat16, float>(TensorLayout::NCHW);
116 allPassed &= f.template operator()<float, float>(TensorLayout::NHWC);
117 allPassed &= f.template operator()<half, float>(TensorLayout::NHWC);
118 allPassed &= f.template operator()<hip_bfloat16, float>(TensorLayout::NHWC);
119 return allPassed;
120}
