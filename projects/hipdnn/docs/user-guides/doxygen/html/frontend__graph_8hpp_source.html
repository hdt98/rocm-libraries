158 Error initializeEngineConfig(hipdnnBackendDescriptor_t engineHeuristicDesc)
159 {
160 std::vector<std::unique_ptr<detail::ScopedHipdnnBackendDescriptor>> engineConfigs;
161 std::vector<int64_t> engineIds;
162 auto defaultEngineId = getDefaultEngineId();
163 HIPDNN_CHECK_ERROR(hipdnn_frontend::detail::getEngineConfigs(
164 engineConfigs,
165 engineIds,
166 engineHeuristicDesc,
167 _preferredEngineId.has_value() || defaultEngineId.has_value()));
168
169 // Select engine config based on preferred ID or use first available
170 size_t selectedIndex = 0;
171 if(defaultEngineId)
172 {
173 auto defaultId = defaultEngineId.value();
174 auto it = std::find(engineIds.begin(), engineIds.end(), defaultId);
175 if(it != engineIds.end())
176 {
177 selectedIndex = static_cast<size_t>(std::distance(engineIds.begin(), it));
178 HIPDNN_FE_LOG_INFO("Default engine id " << defaultId
179 << " found, using it for execution plan.");
180 }
181 else
182 {
183 HIPDNN_FE_LOG_INFO("Default engine id "
184 << defaultId << " not found, using top engine config instead.");
185 }
186 }
187
188 if(_preferredEngineId.has_value())
189 {
190 bool found = false;
191
192 for(size_t i = 0; i < engineIds.size(); ++i)
193 {
194
195 if(engineIds[i] == _preferredEngineId.value())
196 {
197 selectedIndex = i;
198 found = true;
199 break;
200 }
201 }
202
203 if(!found)
204 {
205 HIPDNN_FE_LOG_WARN("Preferred engine id "
206 << _preferredEngineId.value()
207 << " not found, using top engine config instead.");
208 }
209 }
210
211 HIPDNN_FE_LOG_INFO("Selected engine id " << engineIds[selectedIndex]
212 << " for execution plan.");
213 _engineConfigDesc = std::move(engineConfigs[selectedIndex]);
214
215 return {ErrorCode::OK, ""};
216 }

253 {
254 auto inputs = _sub_nodes[inputNodeIndex]->getNodeInputTensorAttributes();
255 for(auto& input : inputs)
256 {
257 auto it = tensorToOriginNode.find(input);
258 if(it != tensorToOriginNode.end())
259 {
260 size_t outputNodeIndex = it->second;
261 structure.adjacencyList[outputNodeIndex].push_back(inputNodeIndex);
262 }
263 }
264 }

405 std::pair<std::unordered_set<std::shared_ptr<TensorAttributes>>,
406 std::unordered_set<std::shared_ptr<TensorAttributes>>>
407 getGraphInputTensorAttributesAndRemainder() const
408 {
409 std::unordered_set<std::shared_ptr<TensorAttributes>> allNodeOutputs;
410 std::unordered_set<std::shared_ptr<TensorAttributes>> graphInputs;
411
412 auto collectNodeOutputs = [&](const INode& node) {
413 auto nodeOutputs = node.getNodeOutputTensorAttributes();
414 allNodeOutputs.insert(nodeOutputs.begin(), nodeOutputs.end());
415 };
416 auto collectGraphInputs = [&](const INode& node) {
417 auto nodeInputs = node.getNodeInputTensorAttributes();
418 std::copy_if(nodeInputs.begin(),
419 nodeInputs.end(),
420 std::inserter(graphInputs, graphInputs.end()),
421 [&](const auto& nodePtr) { return allNodeOutputs.count(nodePtr) == 0; });
422 };
423
424 visit(collectNodeOutputs);
425 visit(collectGraphInputs);
426
427 return {graphInputs, allNodeOutputs};
428 }

431 {
432 std::unordered_set<std::shared_ptr<TensorAttributes>> allTensors;

434

464
465 builder.Finish(graph);

469 Error deserializeFromFlatBuffer(const hipdnn_data_sdk::data_objects::Graph* fbGraph)
470 {
471 // Set graph attributes from FlatBuffer
472 if(fbGraph->name() != nullptr)
473 {
474 set_name(fbGraph->name()->c_str());
475 }
476 set_compute_data_type(fromSdkType(fbGraph->compute_data_type()));
477 set_intermediate_data_type(fromSdkType(fbGraph->intermediate_data_type()));
478 set_io_data_type(fromSdkType(fbGraph->io_data_type()));
479
480 if(fbGraph->preferred_engine_id().has_value())
481 {
482 _preferredEngineId = fbGraph->preferred_engine_id().value();
483 }
484
485 // Build tensorMap from FlatBuffer tensors
486 std::unordered_map<int64_t, std::shared_ptr<TensorAttributes>> tensorMap;
487 if(fbGraph->tensors() != nullptr)
488 {
489 for(const auto* fbTensor : *fbGraph->tensors())
490 {
491 auto tensor = TensorAttributes::fromFlatBuffer(fbTensor);
492 if(tensor != nullptr && tensor->has_uid())
493 {
494 tensorMap[tensor->get_uid()] = tensor;
495 }
496 }
497 }
498
499 // Create nodes from FlatBuffer
500 if(fbGraph->nodes() != nullptr)
501 {
502 for(const auto* fbNode : *fbGraph->nodes())
503 {
504 auto type = fbNode->attributes_type();
505
506 switch(type)
507 {
508 case hipdnn_data_sdk::data_objects::NodeAttributes::BatchnormAttributes:
509 {
510 auto attr = BatchnormAttributes::fromFlatBuffer(
511 fbNode->attributes_as_BatchnormAttributes(), tensorMap);
512 if(fbNode->name() != nullptr)
513 {
514 attr.set_name(fbNode->name()->str());
515 }
516 _sub_nodes.emplace_back(
517 std::make_shared<BatchnormNode>(std::move(attr), graph_attributes));
518 break;
519 }
520 case hipdnn_data_sdk::data_objects::NodeAttributes::BatchnormBackwardAttributes:
521 {
522 auto attr = BatchnormBackwardAttributes::fromFlatBuffer(
523 fbNode->attributes_as_BatchnormBackwardAttributes(), tensorMap);

525 {
526 attr.set_name(fbNode->name()->str());
527 }
528 _sub_nodes.emplace_back(
529 std::make_shared<BatchnormBackwardNode>(std::move(attr), graph_attributes));
530 break;
531 }
532 case hipdnn_data_sdk::data_objects::NodeAttributes::BatchnormInferenceAttributes:
533 {
534 auto attr = BatchnormInferenceAttributes::fromFlatBuffer(
535 fbNode->attributes_as_BatchnormInferenceAttributes(), tensorMap);

537 {
538 attr.set_name(fbNode->name()->str());
539 }
540 _sub_nodes.emplace_back(std::make_shared<BatchnormInferenceNode>(

543 }

744 Error topologicallySortGraph()
745 {
746 size_t nodeCount = _sub_nodes.size();
747
748 if(nodeCount == 0)
749 {
750 return {ErrorCode::OK, ""};
751 }
752
753 auto tensorToOriginNode = buildTensorToOriginNodeMap();
754 auto graphStructure = buildAdjacencyList(tensorToOriginNode);
755
756 auto sortResult = detail::performTopologicalSortWithComponentDetection(graphStructure);
757
758 if(sortResult.hasCycle)
759 {
760 return {ErrorCode::INVALID_VALUE, "Graph contains a cycle, cannot be sorted."};
761 }
762
763 if(sortResult.componentCount > 1)
764 {
765 return {ErrorCode::INVALID_VALUE,
766 "Graph contains multiple disconnected components, please split the graph into "
767 "individual graphs"};
768 }
769
770 reorderNodesTopologically(sortResult.order);
771
772 return {ErrorCode::OK, ""};
773 }

934 Error create_execution_plan_ext(int64_t engineId, const std::vector<KnobSetting>& settings)
935 {
936 HIPDNN_FE_LOG_INFO("Creating execution plans for graph " << graph_attributes.get_name());
937
938 if(!_graphDesc || !_graphDesc->valid())
939 {
940 return {ErrorCode::HIPDNN_BACKEND_ERROR,
941 "Graph has not been built, build the operation graph first. Cannot create "
942 "execution plan."};
943 }
944
945 std::unordered_map<KnobType_t, Knob> existingKnobs;
946 HIPDNN_CHECK_ERROR(get_knob_lookup_for_engine(engineId, existingKnobs));
947 HIPDNN_CHECK_ERROR(initializeEngineConfig(engineId));
948
949 std::vector<KnobSetting> validatedSettings;
950 for(const auto& setting : settings)
951 {
952 auto knobIt = existingKnobs.find(setting.knobId());
953 if(knobIt == existingKnobs.end())
954 {
955 HIPDNN_FE_LOG_WARN("Ignoring knob " << setting.knobId()
956 << " when creating execution plan for graph "
957 << graph_attributes.get_name()
958 << ". Engine doesn't support chosen knob.");
959 continue;
960 }
961
962 const auto& knob = knobIt->second;
963
964 if(knob.isDeprecated())
965 {
966 HIPDNN_FE_LOG_WARN("Knob " << knob.knobId() << " has been marked as deprecated.");
967 }
968
969 HIPDNN_CHECK_ERROR(knob.validate(setting));
970
971 validatedSettings.emplace_back(setting);
972 }
973
974 if(!validatedSettings.empty())
975 {
976 std::vector<flatbuffers::DetachedBuffer> knobBuffers;
977 knobBuffers.reserve(validatedSettings.size());
978
979 for(const auto& setting : validatedSettings)
980 {
981 flatbuffers::FlatBufferBuilder builder;
982 auto knobSettingOffset = setting.packKnobSetting(builder);
983 builder.Finish(knobSettingOffset);
984 knobBuffers.push_back(builder.Release());
985 }
986
987 std::vector<hipdnnBackendFlatbufferData_t> flatbufferDataArray;
988 flatbufferDataArray.reserve(knobBuffers.size());
989
990 for(const auto& buffer : knobBuffers)
991 {
992 hipdnnBackendFlatbufferData_t fbData;
993 fbData.ptr = buffer.data();
994 fbData.size = buffer.size();
995 flatbufferDataArray.push_back(fbData);
996 }
997
998 HIPDNN_RETURN_ON_BACKEND_FAILURE(detail::hipdnnBackend()->backendSetAttribute(
999 _engineConfigDesc->get(),
1000 HIPDNN_ATTR_KNOB_CHOICE_SERIALIZED_VALUE_EXT,
1001 HIPDNN_TYPE_FLATBUFFER_DATA_STRUCT_EXT,
1002 static_cast<int64_t>(flatbufferDataArray.size()),
1003 flatbufferDataArray.data()),
1004 "Failed to set knob settings on engine config.");
1005 }
1006
1007 // Finalize engine config after knobs have been set
1008 HIPDNN_RETURN_ON_BACKEND_FAILURE(
1009 detail::hipdnnBackend()->backendFinalize(_engineConfigDesc->get()),
1010 "Failed to finalize engine config descriptor");
1011
1012 // Create execution plan descriptor
1013 _executionPlanDesc = std::make_unique<detail::ScopedHipdnnBackendDescriptor>(
1014 HIPDNN_BACKEND_EXECUTION_PLAN_DESCRIPTOR);
1015
1016 if(!_executionPlanDesc->valid())
1017 {
1018 return {ErrorCode::HIPDNN_BACKEND_ERROR,
1019 "Failed to create backend execution descriptor."};
1020 }
1021
1022 return {ErrorCode::OK, ""};
1023 }
