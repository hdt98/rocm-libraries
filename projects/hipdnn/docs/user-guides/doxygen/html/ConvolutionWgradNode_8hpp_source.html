28 Error pre_validate_node() const override
29 {
30 auto x = attributes.get_x();
31 auto dy = attributes.get_dy();
32 auto dw = attributes.get_dw();
33
34 HIPDNN_RETURN_IF_FALSE(x,
35 ErrorCode::ATTRIBUTE_NOT_SET,
36 "ConvolutionWgradNode missing x (input) for pre-validation");
37
38 HIPDNN_RETURN_IF_FALSE(
39 dy,
40 ErrorCode::ATTRIBUTE_NOT_SET,
41 "ConvolutionWgradNode missing dy (gradient of output) for pre-validation");
42
43 HIPDNN_RETURN_IF_FALSE(
44 dw,
45 ErrorCode::ATTRIBUTE_NOT_SET,
46 "ConvolutionWgradNode missing dw (gradient of weights) for pre-validation");
47
48 auto& xDims = x->get_dim();
49 auto& dyDims = dy->get_dim();
50 auto& dwDims = dw->get_dim();
51
52 auto spatialDims = dyDims.size() - 2; // N & C dimensions aren't spatial
53 auto& prePadding = attributes.get_pre_padding();
54 auto& postPadding = attributes.get_post_padding();
55 auto& stride = attributes.get_stride();
56 auto& dilation = attributes.get_dilation();
57
58 HIPDNN_RETURN_IF_TRUE(attributes.get_pre_padding().empty(),
59 ErrorCode::ATTRIBUTE_NOT_SET,
60 "ConvolutionWgradNode missing pre_padding for pre-validation");
61
62 HIPDNN_RETURN_IF_TRUE(attributes.get_post_padding().empty(),
63 ErrorCode::ATTRIBUTE_NOT_SET,
64 "ConvolutionWgradNode missing post_padding for pre-validation");
65
66 HIPDNN_RETURN_IF_TRUE(attributes.get_stride().empty(),
67 ErrorCode::ATTRIBUTE_NOT_SET,
68 "ConvolutionWgradNode missing stride for pre-validation");
69
70 HIPDNN_RETURN_IF_TRUE(attributes.get_dilation().empty(),
71 ErrorCode::ATTRIBUTE_NOT_SET,
72 "ConvolutionWgradNode missing dilation for pre-validation");
73
74 // dy implicitly checked here too since they must be equal
75 HIPDNN_RETURN_IF_LT(
76 xDims.size(),
77 3,
78 ErrorCode::INVALID_VALUE,
79 "ConvolutionWgradNode: x tensor must have at least 3 dimensions (N, C, spatial)");
80
81 HIPDNN_RETURN_IF_NE(dyDims.size(),
82 xDims.size(),
83 ErrorCode::INVALID_VALUE,
84 "ConvolutionWgradNode: dy tensor dimension count must match x tensor "
85 "dimension count");
86
87 HIPDNN_RETURN_IF_NE(
88 xDims[0],
89 dyDims[0],
90 ErrorCode::INVALID_VALUE,
91 "ConvolutionWgradNode: x tensor batch size must match dy tensor batch size");
92
93 if(!dwDims.empty())
94 {
95 HIPDNN_RETURN_IF_NE(
96 dwDims.size(),
97 dyDims.size(),
98 ErrorCode::INVALID_VALUE,
99 "ConvolutionWgradNode: dw tensor dimension count must match dy tensor "
100 "dimension count");
101
102 // Validate output channels match between dy and dw tensors
103 HIPDNN_RETURN_IF_NE(
104 dyDims[1],
105 dwDims[0],
106 ErrorCode::INVALID_VALUE,
107 "ConvolutionWgradNode: dy tensor channels must match dw tensor output channels");
108
109 HIPDNN_RETURN_IF_NE(xDims[1] % dwDims[1],
110 0,
111 ErrorCode::INVALID_VALUE,
112 "ConvolutionWgradNode: x tensor channels must be divisible by dw "
113 "tensor input channels");
114
115 // xDims[1] / dwDims[1] is group count
116 auto groupCount = xDims[1] / dwDims[1];
117 HIPDNN_RETURN_IF_NE(
118 dwDims[0] % groupCount,
119 0,
120 ErrorCode::INVALID_VALUE,
121 "ConvolutionWgradNode: dw tensor output channels must be divisible by "
122 "the number of groups");
123
124 // Verifies that spatial dimensions are compatible
125 for(size_t i = 0; i < spatialDims; ++i)
126 {
127 auto spatialIdx = i + 2;
128 int64_t xDim = xDims[spatialIdx];
129 int64_t dyDim = dyDims[spatialIdx];
130 int64_t kernelDim = dwDims[spatialIdx];
131
132 int64_t kernelSize = (dilation[i] * (kernelDim - 1)) + 1;
133 auto numerator = xDim + prePadding[i] + postPadding[i] - kernelSize;
134
135 HIPDNN_RETURN_IF_LT(numerator,
136 0,
137 ErrorCode::INVALID_VALUE,
138 "ConvolutionWgradNode: Input spatial dimension at index "
139 + std::to_string(i) + " (" + std::to_string(xDim)
140 + ") is too small for the kernel size ("
141 + std::to_string(kernelDim) + ") and dilation ("
142 + std::to_string(dilation[i]) + ")");
143
144 int64_t expectedDyDim = (numerator / stride[i]) + 1;
145
146 // Verifying dy implicitly verifies dw and x
147 HIPDNN_RETURN_IF_NE(
148 dyDim,
149 expectedDyDim,
150 ErrorCode::INVALID_VALUE,
151 "ConvolutionWgradNode: dy tensor spatial dimension at index "
152 + std::to_string(i) + " (" + std::to_string(dyDim)
153 + ") does not match expected dimension (" + std::to_string(expectedDyDim)
154 + ") given x dimensions, kernel size, padding, stride, and dilation");
155 }
156 }
157
158 HIPDNN_RETURN_IF_NE(
159 prePadding.size(),
160 spatialDims,
161 ErrorCode::INVALID_VALUE,
162 "ConvolutionWgradNode: pre_padding parameter count must match spatial dimension count");
163
164 HIPDNN_RETURN_IF_NE(postPadding.size(),
165 spatialDims,
166 ErrorCode::INVALID_VALUE,
167 "ConvolutionWgradNode: post_padding parameter count must match spatial "
168 "dimension count");
169
170 HIPDNN_RETURN_IF_NE(
171 stride.size(),
172 spatialDims,
173 ErrorCode::INVALID_VALUE,
174 "ConvolutionWgradNode: stride parameter count must match spatial dimension count");
175
176 HIPDNN_RETURN_IF_NE(
177 dilation.size(),
178 spatialDims,
179 ErrorCode::INVALID_VALUE,
180 "ConvolutionWgradNode: dilation parameter count must match spatial dimension count");
181
182 // Check spatial parameters for each dimension
183 for(size_t i = 0; i < spatialDims; ++i)
184 {
185 auto prePad = prePadding[i];
186 auto postPad = postPadding[i];
187 auto strideVal = stride[i];
188 auto dilationVal = dilation[i];
189
190 HIPDNN_RETURN_IF_LT(
191 strideVal, 1, ErrorCode::INVALID_VALUE, "ConvolutionWgradNode: Stride must be > 0");
192
193 HIPDNN_RETURN_IF_LT(dilationVal,
194 1,
195 ErrorCode::INVALID_VALUE,
196 "ConvolutionWgradNode: Dilation must > 0");
197
198 HIPDNN_RETURN_IF_LT(prePad,
199 0,
200 ErrorCode::INVALID_VALUE,
201 "ConvolutionWgradNode: Pre-padding must be non-negative");
202
203 HIPDNN_RETURN_IF_LT(postPad,
204 0,
205 ErrorCode::INVALID_VALUE,
206 "ConvolutionWgradNode: Post-padding must be non-negative");
207 }
208
209 return {};
210 }

212 Error infer_properties_node() override
213 {
214 auto x = attributes.get_x();
215 auto dy = attributes.get_dy();
216 auto dw = attributes.get_dw();
217
218 // Repeated checks from pre_validate_node for cases where this is called standalone
219 HIPDNN_RETURN_IF_FALSE(x,
220 ErrorCode::ATTRIBUTE_NOT_SET,
221 "ConvolutionWgradNode missing x for setting properties");
222
223 HIPDNN_RETURN_IF_FALSE(dy,
224 ErrorCode::ATTRIBUTE_NOT_SET,
225 "ConvolutionWgradNode missing dy for setting properties");
226
227 HIPDNN_RETURN_IF_FALSE(dw,
228 ErrorCode::ATTRIBUTE_NOT_SET,
229 "ConvolutionWgradNode missing dw for setting properties");
230
231 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
232
233 auto dwDims = dw->get_dim();
234
235 if(dwDims.empty())
236 {
237 auto& xDims = x->get_dim();
238 auto& dyDims = dy->get_dim();
239
240 dwDims.resize(dyDims.size());
241
242 auto& prePadding = attributes.get_pre_padding();
243 auto& postPadding = attributes.get_post_padding();
244 auto& stride = attributes.get_stride();
245 auto& dilation = attributes.get_dilation();
246
247 dwDims[0] = dyDims[1]; // Output channels match dy channels
248
249 // Impossible to infer group count without dw dimensions.
250 // Therefore, assume groups = 1.
251 dwDims[1] = xDims[1]; // Input channels (per group)
252
253 // Calculate kernel spatial dimensions (k_2, ..., k_n)
254 for(size_t i = 2; i < dyDims.size(); ++i)
255 {
256 auto spatialIdx = i - 2;
257
258 HIPDNN_RETURN_IF_TRUE(
259 spatialIdx >= prePadding.size() || spatialIdx >= postPadding.size()
260 || spatialIdx >= stride.size() || spatialIdx >= dilation.size(),
261 ErrorCode::INVALID_VALUE,
262 "ConvolutionWgradNode: Insufficient padding/stride/dilation parameters for "
263 "spatial dimensions");
264
265 auto xSize = xDims[i];
266 auto dySize = dyDims[i];
267 auto prePad = prePadding[spatialIdx];
268 auto postPad = postPadding[spatialIdx];
269 auto strideVal = stride[spatialIdx];
270 auto dilationVal = dilation[spatialIdx];
271
272 HIPDNN_RETURN_IF_LT(strideVal,
273 1,
274 ErrorCode::INVALID_VALUE,
275 "ConvolutionWgradNode: Stride must be positive");
276
277 HIPDNN_RETURN_IF_LT(dilationVal,
278 1,
279 ErrorCode::INVALID_VALUE,
280 "ConvolutionWgradNode: Dilation must be positive");
281
282 HIPDNN_RETURN_IF_LT(prePad,
283 0,
284 ErrorCode::INVALID_VALUE,
285 "ConvolutionWgradNode: Pre-padding must be non-negative");
286
287 HIPDNN_RETURN_IF_LT(postPad,
288 0,
289 ErrorCode::INVALID_VALUE,
290 "ConvolutionWgradNode: Post-padding must be non-negative");
291
292 // Conv fwd output spatial dim size:
293 // out_i = (in_i + pre_padding + post_padding - dilation * (kernel_size - 1) - 1) / stride + 1
294
295 // Solve for kernel size:
296 // kernel_size = ((in_i + pre_padding + post_padding - stride * (out_i - 1) - 1) / dilation) + 1
297 auto numerator = xSize + prePad + postPad - (strideVal * (dySize - 1)) - 1;
298
299 HIPDNN_RETURN_IF_LT(
300 numerator,
301 0,
302 ErrorCode::INVALID_VALUE,
303 "ConvolutionWgradNode: Invalid spatial dimensions for kernel size inference");
304
305 // Calculate the remainder of pixels that are "dropped" at the end of the convolution
306 // We want to find the smallest remainder r such that (numerator - r) is divisible by dilation
307 // and r < stride.
308 // r = numerator % dilation satisfies the divisibility.
309 // We check if it satisfies the stride constraint.
310 auto remainder = numerator % dilationVal;
311
312 HIPDNN_RETURN_IF_GE(remainder,
313 strideVal,
314 ErrorCode::INVALID_VALUE,
315 "ConvolutionWgradNode: Spatial dimensions incompatible with "
316 "dilation and stride parameters for kernel size inference");
317
318 dwDims[i] = ((numerator - remainder) / dilationVal) + 1;
319 }
320
321 dw->set_dim(dwDims);
322 }
323
324 if(dw->get_stride().empty())
325 {
326 auto& xStrides = x->get_stride();
327 auto& dwDimsFinal = dw->get_dim();
328
329 HIPDNN_RETURN_IF_TRUE(
330 xStrides.empty(),
331 ErrorCode::ATTRIBUTE_NOT_SET,
332 "ConvolutionWgradNode: Cannot infer dw strides - missing x strides");
333
334 HIPDNN_RETURN_IF_NE(
335 xStrides.size(),
336 dwDimsFinal.size(),
337 ErrorCode::ATTRIBUTE_NOT_SET,
338 "ConvolutionWgradNode: Stride dimension mismatch between x and dw tensors");
339
340 // Extract stride order from x tensor and apply to dw tensor
341 auto strideOrder = hipdnn_data_sdk::utilities::extractStrideOrder(xStrides);
342
343 // Generate dw strides using the extracted stride order and dw dimensions
344 auto dwStrides = hipdnn_data_sdk::utilities::generateStrides(dwDimsFinal, strideOrder);
345
346 dw->set_stride(dwStrides);
347 }
348
349 return {};
350 }

360 attributes.pack_attributes(builder).Union());
361 }
