186template <typename T>
187DataType getDataTypeEnumFromType()
188{
189 if constexpr(std::is_same_v<T, float>)
190 {
191 return DataType::FLOAT;
192 }
193 else if constexpr(std::is_same_v<T, half>)
194 {
195 return DataType::HALF;
196 }
197 else if constexpr(std::is_same_v<T, bfloat16>)
198 {
199 return DataType::BFLOAT16;
200 }
201 else if constexpr(std::is_same_v<T, double>)
202 {
203 return DataType::DOUBLE;
204 }
205 else if constexpr(std::is_same_v<T, uint8_t>)
206 {
207 return DataType::UINT8;
208 }
209 else if constexpr(std::is_same_v<T, int32_t>)
210 {
211 return DataType::INT32;
212 }
213 else if constexpr(std::is_same_v<T, int8_t>)
214 {
215 return DataType::INT8;
216 }
217 else if constexpr(std::is_same_v<T, fp8_e4m3>)
218 {
219 return DataType::FP8_E4M3;
220 }
221 else if constexpr(std::is_same_v<T, fp8_e5m2>)
222 {
223 return DataType::FP8_E5M2;
224 }
225 else
226 {
227 return DataType::NOT_SET;
228 }
229}
