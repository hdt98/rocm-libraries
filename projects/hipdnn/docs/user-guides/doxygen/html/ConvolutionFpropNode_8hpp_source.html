26 Error pre_validate_node() const override
27 {
28 // Validate tensor pointers
29 HIPDNN_RETURN_IF_FALSE(attributes.get_x(),
30 ErrorCode::ATTRIBUTE_NOT_SET,
31 "ConvolutionFpropNode missing x (input) for pre-validation");
32
33 HIPDNN_RETURN_IF_FALSE(attributes.get_w(),
34 ErrorCode::ATTRIBUTE_NOT_SET,
35 "ConvolutionFpropNode missing w (weights) for pre-validation");
36
37 HIPDNN_RETURN_IF_FALSE(attributes.get_y(),
38 ErrorCode::ATTRIBUTE_NOT_SET,
39 "ConvolutionFpropNode missing y (output) for pre-validation");
40
41 // Validate convolution parameters
42 HIPDNN_RETURN_IF_TRUE(attributes.get_pre_padding().empty(),
43 ErrorCode::ATTRIBUTE_NOT_SET,
44 "ConvolutionFpropNode missing pre_padding for pre-validation");
45
46 HIPDNN_RETURN_IF_TRUE(attributes.get_post_padding().empty(),
47 ErrorCode::ATTRIBUTE_NOT_SET,
48 "ConvolutionFpropNode missing post_padding for pre-validation");
49
50 HIPDNN_RETURN_IF_TRUE(attributes.get_stride().empty(),
51 ErrorCode::ATTRIBUTE_NOT_SET,
52 "ConvolutionFpropNode missing stride for pre-validation");
53
54 HIPDNN_RETURN_IF_TRUE(attributes.get_dilation().empty(),
55 ErrorCode::ATTRIBUTE_NOT_SET,
56 "ConvolutionFpropNode missing dilation for pre-validation");
57
58 // Get tensor references
59 auto x = attributes.get_x();
60 auto w = attributes.get_w();
61 auto y = attributes.get_y();
62
63 // Validate input tensor dimensions and strides
64 auto& xDims = x->get_dim();
65
66 HIPDNN_RETURN_IF_LT(
67 xDims.size(),
68 3,
69 ErrorCode::INVALID_VALUE,
70 "ConvolutionFpropNode: Input tensor must have at least 3 dimensions (N, C, spatial)");
71
72 // Validate weight tensor dimensions and strides
73 auto& wDims = w->get_dim();
74
75 HIPDNN_RETURN_IF_NE(
76 wDims.size(),
77 xDims.size(),
78 ErrorCode::INVALID_VALUE,
79 "ConvolutionFpropNode: Weight tensor dimension count must match input tensor "
80 "dimension count");
81
82 // Validate input channels match between input and weight tensors
83 // For regular convolution: x_dims[1] == w_dims[1]
84 // For grouped convolution: x_dims[1] % w_dims[1] == 0
85 HIPDNN_RETURN_IF_NE(
86 xDims[1] % wDims[1],
87 0,
88 ErrorCode::INVALID_VALUE,
89 "ConvolutionFpropNode: Input tensor channels must match weight tensor input "
90 "channels or be divisible by them for grouped convolution");
91
92 // For grouped convolution: x_dims[1] / w_dims[1] is group count.
93 // Output channels must be divisible by group count.
94 auto groupCount = xDims[1] / wDims[1];
95 HIPDNN_RETURN_IF_NE(
96 wDims[0] % groupCount,
97 0,
98 ErrorCode::INVALID_VALUE,
99 "ConvolutionFpropNode: Weight tensor output channels must be divisible by "
100 "the number of groups");
101
102 // Validate output tensor dimensions and strides if they are set
103 auto& yDims = y->get_dim();
104
105 if(!yDims.empty())
106 {
107 HIPDNN_RETURN_IF_NE(
108 yDims.size(),
109 xDims.size(),
110 ErrorCode::INVALID_VALUE,
111 "ConvolutionFpropNode: Output tensor dimension count must match input tensor "
112 "dimension count");
113
114 // Validate batch size matches
115 HIPDNN_RETURN_IF_NE(yDims[0],
116 xDims[0],
117 ErrorCode::INVALID_VALUE,
118 "ConvolutionFpropNode: Output tensor batch size must match input "
119 "tensor batch size");
120
121 // Validate output channels match weight output channels
122 HIPDNN_RETURN_IF_NE(yDims[1],
123 wDims[0],
124 ErrorCode::INVALID_VALUE,
125 "ConvolutionFpropNode: Output tensor channels must match weight "
126 "tensor output channels");
127 }
128
129 // Validate spatial parameter counts match spatial dimensions
130 auto spatialDims = xDims.size() - 2; // Skip N and C dimensions
131 auto& prePadding = attributes.get_pre_padding();
132 auto& postPadding = attributes.get_post_padding();
133 auto& stride = attributes.get_stride();
134 auto& dilation = attributes.get_dilation();
135
136 HIPDNN_RETURN_IF_NE(
137 prePadding.size(),
138 spatialDims,
139 ErrorCode::INVALID_VALUE,
140 "ConvolutionFpropNode: pre_padding parameter count must match spatial dimension count");
141
142 HIPDNN_RETURN_IF_NE(postPadding.size(),
143 spatialDims,
144 ErrorCode::INVALID_VALUE,
145 "ConvolutionFpropNode: post_padding parameter count must match spatial "
146 "dimension count");
147
148 HIPDNN_RETURN_IF_NE(
149 stride.size(),
150 spatialDims,
151 ErrorCode::INVALID_VALUE,
152 "ConvolutionFpropNode: stride parameter count must match spatial dimension count");
153
154 HIPDNN_RETURN_IF_NE(
155 dilation.size(),
156 spatialDims,
157 ErrorCode::INVALID_VALUE,
158 "ConvolutionFpropNode: dilation parameter count must match spatial dimension count");
159
160 // Check spatial parameters for each dimension
161 for(size_t i = 0; i < spatialDims; ++i)
162 {
163 auto prePad = prePadding[i];
164 auto postPad = postPadding[i];
165 auto strideVal = stride[i];
166 auto dilationVal = dilation[i];
167
168 // Validate parameters
169 HIPDNN_RETURN_IF_LT(
170 strideVal, 1, ErrorCode::INVALID_VALUE, "ConvolutionFpropNode: Stride must be > 0");
171
172 HIPDNN_RETURN_IF_LT(dilationVal,
173 1,
174 ErrorCode::INVALID_VALUE,
175 "ConvolutionFpropNode: Dilation must > 0");
176
177 HIPDNN_RETURN_IF_LT(prePad,
178 0,
179 ErrorCode::INVALID_VALUE,
180 "ConvolutionFpropNode: Pre-padding must be non-negative");
181
182 HIPDNN_RETURN_IF_LT(postPad,
183 0,
184 ErrorCode::INVALID_VALUE,
185 "ConvolutionFpropNode: Post-padding must be non-negative");
186
187 if(!yDims.empty())
188 {
189 auto inputSize = xDims[i + 2];
190 auto kernelSize = wDims[i + 2];
191 auto outputSize = yDims[i + 2];
192
193 auto dilatedKernelSize = (dilationVal * (kernelSize - 1)) + 1;
194 auto numerator = inputSize + prePad + postPad - dilatedKernelSize;
195
196 HIPDNN_RETURN_IF_LT(numerator,
197 0,
198 ErrorCode::INVALID_VALUE,
199 "ConvolutionFpropNode: Input spatial dimension at index "
200 + std::to_string(i) + " (" + std::to_string(inputSize)
201 + ") is too small for the kernel size ("
202 + std::to_string(kernelSize) + ") and dilation ("
203 + std::to_string(dilationVal) + ")");
204
205 int64_t expectedOutputSize = (numerator / strideVal) + 1;
206
207 HIPDNN_RETURN_IF_NE(
208 outputSize,
209 expectedOutputSize,
210 ErrorCode::INVALID_VALUE,
211 "ConvolutionFpropNode: Output tensor spatial dimension at index "
212 + std::to_string(i) + " (" + std::to_string(outputSize)
213 + ") does not match expected dimension ("
214 + std::to_string(expectedOutputSize)
215 + ") given input dimensions, kernel size, padding, stride, and dilation");
216 }
217 }
218
219 return {};
220 }

222 Error infer_properties_node() override
223 {
224 auto x = attributes.get_x();
225 auto w = attributes.get_w();
226 auto y = attributes.get_y();
227
228 HIPDNN_RETURN_IF_FALSE(x,
229 ErrorCode::ATTRIBUTE_NOT_SET,
230 "ConvolutionFpropNode missing x for setting properties");
231
232 HIPDNN_RETURN_IF_FALSE(w,
233 ErrorCode::ATTRIBUTE_NOT_SET,
234 "ConvolutionFpropNode missing w for setting properties");
235
236 HIPDNN_RETURN_IF_FALSE(y,
237 ErrorCode::ATTRIBUTE_NOT_SET,
238 "ConvolutionFpropNode missing y for setting properties");
239
240 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
241
242 auto yDims = y->get_dim();
243
244 // Infer output dimensions if not set
245 if(yDims.empty())
246 {
247 auto& xDims = x->get_dim();
248 auto& wDims = w->get_dim();
249
250 yDims.resize(xDims.size());
251
252 auto& prePadding = attributes.get_pre_padding();
253 auto& postPadding = attributes.get_post_padding();
254 auto& stride = attributes.get_stride();
255 auto& dilation = attributes.get_dilation();
256
257 yDims[0] = xDims[0]; // N (batch) matches input
258 yDims[1] = wDims[0]; // C (output channels)
259
260 // Calculate spatial dimensions (Optional D, H, W)
261 // Starting from dim 2 (skip N and C)
262 for(size_t i = 2; i < xDims.size(); ++i)
263 {
264 auto spatialIdx = i - 2; // Index into spatial dimension arrays
265
266 HIPDNN_RETURN_IF_TRUE(
267 spatialIdx >= prePadding.size() || spatialIdx >= postPadding.size()
268 || spatialIdx >= stride.size() || spatialIdx >= dilation.size(),
269 ErrorCode::INVALID_VALUE,
270 "ConvolutionFpropNode: Insufficient padding/stride/dilation parameters for "
271 "spatial "
272 "dimensions");
273
274 // Standard convolution output size formula:
275 // output_size = floor((input_size + pre_padding + post_padding - dilated_kernel_size) / stride) + 1
276 // where dilated_kernel_size = dilation * (kernel_size - 1) + 1
277
278 auto inputSize = xDims[i];
279 auto kernelSize = wDims[i];
280 auto prePad = prePadding[spatialIdx];
281 auto postPad = postPadding[spatialIdx];
282 auto strideVal = stride[spatialIdx];
283 auto dilationVal = dilation[spatialIdx];
284
285 // Validate parameters
286 HIPDNN_RETURN_IF_LT(strideVal,
287 1,
288 ErrorCode::INVALID_VALUE,
289 "ConvolutionFpropNode: Stride must be positive");
290
291 HIPDNN_RETURN_IF_LT(dilationVal,
292 1,
293 ErrorCode::INVALID_VALUE,
294 "ConvolutionFpropNode: Dilation must be positive");
295
296 HIPDNN_RETURN_IF_LT(prePad,
297 0,
298 ErrorCode::INVALID_VALUE,
299 "ConvolutionFpropNode: Pre-padding must be non-negative");
300
301 HIPDNN_RETURN_IF_LT(postPad,
302 0,
303 ErrorCode::INVALID_VALUE,
304 "ConvolutionFpropNode: Post-padding must be non-negative");
305
306 // Calculate dilated kernel size
307 auto dilatedKernelSize = (dilationVal * (kernelSize - 1)) + 1;
308
309 // Calculate output dimension
310 auto numerator = inputSize + prePad + postPad - dilatedKernelSize;
311 HIPDNN_RETURN_IF_LT(
312 numerator,
313 0,
314 ErrorCode::INVALID_VALUE,
315 "ConvolutionFpropNode: Invalid convolution parameters result in negative "
316 "output size");
317
318 yDims[i] = (numerator / strideVal) + 1;
319 }
320
321 // Set the inferred dimensions
322 y->set_dim(yDims);
323 }
324
325 // Infer output strides if not set
326 if(y->get_stride().empty())
327 {
328 auto& xStrides = x->get_stride();
329 auto& yDimsFinal = y->get_dim();
330
331 HIPDNN_RETURN_IF_TRUE(
332 xStrides.empty(),
333 ErrorCode::ATTRIBUTE_NOT_SET,
334 "ConvolutionFpropNode: Cannot infer output strides - missing input strides");
335
336 HIPDNN_RETURN_IF_TRUE(
337 yDimsFinal.empty(),
338 ErrorCode::ATTRIBUTE_NOT_SET,
339 "ConvolutionFpropNode: Cannot infer output strides - missing output dimensions");
340
341 HIPDNN_RETURN_IF_NE(
342 xStrides.size(),
343 yDimsFinal.size(),
344 ErrorCode::ATTRIBUTE_NOT_SET,
345 "ConvolutionFpropNode: Stride dimension mismatch between input and output tensors");
346
347 // Extract stride order from input tensor and apply to output tensor
348 auto strideOrder = hipdnn_data_sdk::utilities::extractStrideOrder(xStrides);
349
350 // Generate Y strides using the extracted stride order and Y dimensions
351 auto yStrides = hipdnn_data_sdk::utilities::generateStrides(yDimsFinal, strideOrder);
352
353 y->set_stride(yStrides);
354 }
355
356 return {};
357 }

368 }
