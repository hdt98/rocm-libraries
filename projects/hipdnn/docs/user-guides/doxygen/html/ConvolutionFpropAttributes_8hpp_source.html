261 static ConvFpropAttributes fromFlatBuffer(
262 const hipdnn_data_sdk::data_objects::ConvolutionFwdAttributes* fb,
263 const std::unordered_map<int64_t, std::shared_ptr<TensorAttributes>>& tensorMap)
264 {
265 ConvFpropAttributes attr;
266
267 attr.set_x(tensorMap.at(fb->x_tensor_uid()));
268 attr.set_w(tensorMap.at(fb->w_tensor_uid()));
269 attr.set_y(tensorMap.at(fb->y_tensor_uid()));
270
271 if(fb->pre_padding() != nullptr)
272 {
273 std::vector<int64_t> prePadding(fb->pre_padding()->begin(), fb->pre_padding()->end());
274 attr.set_pre_padding(std::move(prePadding));
275 }
276 if(fb->post_padding() != nullptr)
277 {
278 std::vector<int64_t> postPadding(fb->post_padding()->begin(),
279 fb->post_padding()->end());
280 attr.set_post_padding(std::move(postPadding));
281 }
282 if(fb->stride() != nullptr)
283 {
284 std::vector<int64_t> strideVec(fb->stride()->begin(), fb->stride()->end());
285 attr.set_stride(std::move(strideVec));
286 }
287 if(fb->dilation() != nullptr)
288 {
289 std::vector<int64_t> dilationVec(fb->dilation()->begin(), fb->dilation()->end());
290 attr.set_dilation(std::move(dilationVec));
291 }
292 attr.set_convolution_mode(fromSdkType(fb->conv_mode()));
293
294 return attr;
295 }
