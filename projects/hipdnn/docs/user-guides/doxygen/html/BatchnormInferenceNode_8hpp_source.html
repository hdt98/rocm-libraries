26 Error pre_validate_node() const override
27 {
28 // ====================================================================
29 // BATCH NORMALIZATION INFERENCE VALIDATION
30 // (Spatial Mode: per-channel statistics over N×H×W for 4D, N×D×H×W for 5D)
31 // ====================================================================
32 // Algorithm Overview:
33 // During inference, BN uses PRE-COMPUTED running statistics from training.
34 // For each channel c, using saved running stats (runMean_c, runVar_c):
35 //
36 // Normalizes: xhat[n,c,h,w] = (x[n,c,h,w] - runMean_c) / sqrt(runVar_c + ε)
37 // Transforms: y[n,c,h,w] = scale_c * xhat[n,c,h,w] + bias_c
38 //
39 // Key difference from training:
40 // - NO batch statistics computed (no dependence on current batch)
41 // - Uses saved runMean_c and runVar_c (estimated from training)
42 // - NO running stats updates (read-only operation)
43 // ====================================================================
44
45 // SECTION 1: Validate Required Tensor Pointers
46 HIPDNN_RETURN_IF_FALSE(attributes.get_x(),
47 ErrorCode::ATTRIBUTE_NOT_SET,
48 "BatchnormInferenceNode missing x for pre-validation");
49
50 HIPDNN_RETURN_IF_FALSE(attributes.get_scale(),
51 ErrorCode::ATTRIBUTE_NOT_SET,
52 "BatchnormInferenceNode missing scale for pre-validation");
53
54 HIPDNN_RETURN_IF_FALSE(attributes.get_bias(),
55 ErrorCode::ATTRIBUTE_NOT_SET,
56 "BatchnormInferenceNode missing bias for pre-validation");
57
58 HIPDNN_RETURN_IF_FALSE(attributes.get_y(),
59 ErrorCode::ATTRIBUTE_NOT_SET,
60 "BatchnormInferenceNode missing y for pre-validation");
61
62 HIPDNN_RETURN_IF_FALSE(attributes.get_mean(),
63 ErrorCode::ATTRIBUTE_NOT_SET,
64 "BatchnormInferenceNode missing mean for pre-validation");
65
66 HIPDNN_RETURN_IF_FALSE(attributes.get_inv_variance(),
67 ErrorCode::ATTRIBUTE_NOT_SET,
68 "BatchnormInferenceNode missing inv_variance for pre-validation");
69
70 // Get tensor references
71 auto x = attributes.get_x();
72 auto y = attributes.get_y();
73 auto scale = attributes.get_scale();
74 auto bias = attributes.get_bias();
75 auto mean = attributes.get_mean();
76 auto invVar = attributes.get_inv_variance();
77
78 // SECTION 2: Validate Required Tensor Dimensions
79 // Why: All required tensors must have dimensions set by user - they are never inferred.
80 // For inference: x, scale, bias, mean, invVar are all required user parameters.
81 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(x, 2, "Input tensor (x)"));
82 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(scale, 2, "Scale tensor"));
83 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(bias, 2, "Bias tensor"));
84 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(mean, 2, "Mean tensor"));
85 HIPDNN_CHECK_ERROR(validateMinimumTensorDimensions(invVar, 2, "Inverse variance tensor"));
86
87 // SECTION 3: Validate Output Tensor Shape Consistency
88 // Why: BN preserves tensor shape during inference just as in training.
89 // Output y[n,c,h,w] has same shape as input x[n,c,h,w].
90 HIPDNN_CHECK_ERROR(
91 validateTensorShapesMatchIfSet(x, y, "Input tensor (x)", "Output tensor (y)"));
92
93 // SECTION 4: Validate Channel Dimensions and Parameter Tensor Shapes
94 // Why: All parameters are per-channel with shape [1, C, 1, 1]:
95 // - mean_c and var_c: Running statistics saved from training
96 // - scale and bias: Learned parameters from training
97 // These are the same parameters used during training, now fixed for inference.
98
99 // Extract channel count - safe to access xDims[1] after SECTION 2 validation
100 auto& xDims = x->get_dim();
101 int64_t channels = xDims[1];
102
103 // Validate scale has correct channel-only shape (required user parameter)
104 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(scale, channels, "Scale tensor"));
105
106 // Validate bias has correct channel-only shape (required user parameter)
107 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(bias, channels, "Bias tensor"));
108
109 // Validate mean has correct channel-only shape (required user parameter for inference)
110 HIPDNN_CHECK_ERROR(validateChannelOnlyTensorShape(mean, channels, "Mean tensor"));
111
112 // Validate inv_variance has correct channel-only shape (required user parameter for inference)
113 HIPDNN_CHECK_ERROR(
114 validateChannelOnlyTensorShape(invVar, channels, "Inverse variance tensor"));
115
116 // NOTE: Unlike training, inference does NOT require m > 1 (where m = N*H*W for 4D
117 // or m = N*D*H*W for 5D) since it uses pre-computed statistics rather than
118 // computing batch statistics.
119
120 return {ErrorCode::OK, ""};
121 }

123 Error infer_properties_node() override
124 {
125 auto x = attributes.get_x();
126 auto y = attributes.get_y();
127
128 if(!x)
129 {
130 return {ErrorCode::ATTRIBUTE_NOT_SET,
131 "BatchnormInferenceNode missing x for setting properties"};
132 }
133
134 if(!y)
135 {
136 return {ErrorCode::ATTRIBUTE_NOT_SET,
137 "BatchnormInferenceNode missing y for setting properties"};
138 }
139
140 HIPDNN_CHECK_ERROR(attributes.fill_from_context(graph_attributes));
141
142 if(y->get_dim().empty())
143 {
144 y->set_dim(x->get_dim());
145 }
146
147 if(y->get_stride().empty())
148 {
149 y->set_stride(x->get_stride());
150 }
151
152 return {};
153 }

163 attributes.pack_attributes(builder).Union());
164 }
165};
