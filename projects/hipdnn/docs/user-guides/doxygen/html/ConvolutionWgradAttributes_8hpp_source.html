250 static ConvWgradAttributes fromFlatBuffer(
251 const hipdnn_data_sdk::data_objects::ConvolutionWrwAttributes* fb,
252 const std::unordered_map<int64_t, std::shared_ptr<TensorAttributes>>& tensorMap)
253 {
254 ConvWgradAttributes attr;
255
256 attr.set_x(tensorMap.at(fb->x_tensor_uid()));
257 attr.set_dy(tensorMap.at(fb->dy_tensor_uid()));
258 attr.set_dw(tensorMap.at(fb->dw_tensor_uid()));
259
260 if(fb->pre_padding() != nullptr)
261 {
262 std::vector<int64_t> prePadding(fb->pre_padding()->begin(), fb->pre_padding()->end());
263 attr.set_pre_padding(std::move(prePadding));
264 }
265 if(fb->post_padding() != nullptr)
266 {
267 std::vector<int64_t> postPadding(fb->post_padding()->begin(),
268 fb->post_padding()->end());
269 attr.set_post_padding(std::move(postPadding));
270 }
271 if(fb->stride() != nullptr)
272 {
273 std::vector<int64_t> strideVec(fb->stride()->begin(), fb->stride()->end());
274 attr.set_stride(std::move(strideVec));
275 }
276 if(fb->dilation() != nullptr)
277 {
278 std::vector<int64_t> dilationVec(fb->dilation()->begin(), fb->dilation()->end());
279 attr.set_dilation(std::move(dilationVec));
280 }
281 attr.set_convolution_mode(fromSdkType(fb->conv_mode()));
282
283 return attr;
284 }
