15HIPDNN_HIDDEN inline int32_t initializeFrontendLogging(hipdnnCallback_t fn
16 = hipdnnLoggingCallback_ext)
17{
18 if(fn == nullptr)

20 return -1;
21 }
22
23 static bool s_loggingInitialized = false;
24
25 if(s_loggingInitialized)

27 return 0;

31 hipdnn_data_sdk::logging::initializeLogLevel();
32
33 // Register the callback so log messages get routed to the backend
34 hipdnn_data_sdk::logging::registerLoggingCallback(fn);

37
38 // Use this logging macro directly to avoid re-entrant logging call.

41 return 0;
